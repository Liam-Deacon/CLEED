79  - calculate IV curves (program "cleed")

81 
82 ***********************************************************************/
83 {
84  static real rfac_min = 100.;
85  static real rfac_max = 0.;

87  static int n_eval = 0;
88  static int n_calc = 0;

109  ***********************************************************************/

111  n_eval ++;
112  sprintf(log_file, "%s.log", sr_project);
113  sprintf(par_file, "%s.par", sr_project);
114 
115 /***********************************************************************
116  * Check whether environment variables CSEARCH_LEED and CSEARCH_RFAC exist
117  ***********************************************************************/
118

120  {
121  #ifdef ERROR

123  " CSEARCH_LEED or CSEARCH_RFAC not defined\n");
124  #endif
125  exit(SR_ENVIRONMENT_VARIABLE_ERROR);
126  }

128 /***********************************************************************
129  * Geometry assessment
130  ***********************************************************************/
131 
132  #ifdef SHORTCUT
133  fprintf(STDCTR, "(sr_evalrf %d) SHORTCUT:", n_eval);
134  #endif
135 
136  rgeo = sr_ckgeo( par );
137 
138  #ifdef CONTROL
139  #ifdef SHORTCUT
140  fprintf(STDCTR," rgeo = %.4f", rgeo);
141  #else
142  fprintf(STDCTR," rgeo = %.4f\n", rgeo);
143  #endif
144  #endif
145 
146 /***********************************************************************
147  * Return (rfac_max + rgeo), if the test geometry is seriously
148  * outside the limits.
149  * Calculate IV curves otherwise.
150  ***********************************************************************/
151 
152  if( (rgeo > 1.) && (n_calc > 1) )
153  {
154  #ifdef CONTROL
155  fprintf(STDCTR, " return rfac_max: %.4f rtot = %.4f\n",
156  rfac_max, rgeo + rfac_max);
157  #endif
158  log_stream = fopen(log_file, "a");
159 
160  fprintf(log_stream, "#%3d par:", n_eval);
161  for(i_par=1; i_par <= sr_search->n_par; i_par++)
162  fprintf(log_stream, " %.3f", par[i_par]);
163  fprintf(log_stream, " **rfmax: %.4f** rg:%.4f rt:%.4f ",
164  rfac_max, rgeo, rfac_max + rgeo);
165 
166  t_time = time(NULL);
167  l_time = localtime(&t_time);
168  fprintf(log_stream, " dt: %s", asctime(l_time) );
169 
170  fclose(log_stream);
171  return(rfac_max + rgeo);
172  }
173 
174 /***********************************************************************
175  Calculate IV curves
176 ***********************************************************************/
177 
178  n_calc ++;
179  sr_mkinp_mir(par_file, par, n_calc);
180 
181 #ifdef SHORTCUT
182 
183  rfac = 0.;
184  for (i_par = 1; i_par <= sr_search->n_par; i_par ++)
185  rfac += 1 - cos(PI*(par[i_par] -2.3)) +
186  R_fabs(par[i_par] - 2.3);
187  rfac /= sr_search->n_par;
188 
189  #ifdef CONTROL
190  fprintf(STDCTR, " rfac = %.4f rtot = %.4f\n", rfac, rgeo + rfac);
191  #endif
192 
193  rfac_min = MIN(rfac, rfac_min);
194  rfac_max = MAX(rfac, rfac_max);
195 
196 #else
197 
198 /* calculate IV curves for the original geometry */
199  sprintf(line_buffer,
200  "\"%s\" -b \"%s.bul_org\" -i \"%s\" -o \"%s.res_org\" > \"%s.out\"",
201  getenv("CSEARCH_LEED"), /* LEED program name */
202  sr_project, /* project name for bulk file */
203  par_file, /* parameter file for overlayer */
204  sr_project, /* project name for results file */
205  sr_project); /* project name for output file */
206

209  n_eval, line_buffer);
210  #endif
211 
212  if (system (line_buffer)) {SYS_ERROR_TO_LOG(line_buffer);}
213 
214 /* calculate IV curves for the mirrored geometry */
215  sprintf(line_buffer,
216  "\"%s\" -b \"%s.bul_mir\" -i \"%s\" -o \"%s.res_mxx\" > \"%s.out\"",
217  getenv("CSEARCH_LEED"), /* LEED program name */
218  sr_project, /* project name for bulk file */
219  par_file, /* parameter file for overlayer */
220  sr_project, /* project name for results file */
221  sr_project); /* project name for output file */
222 
223 
224  #ifdef CONTROL
225  fprintf(STDCTR,"(sr_evalrf %d): calculate IV curves (mirror):\n %s\n",
226  n_eval, line_buffer);
227  #endif
228 
229  if (system (line_buffer)) {SYS_ERROR_TO_LOG(line_buffer);}
230 
231 /* change indices in IV curves for the mirrored geometry */
232  sprintf(line_buffer,
233  "\"%s/bin/c_miry\" -i \"%s.res_mxx\" -o \"%s.res_mir\" > \"%s.out\"",
234  getenv("CLEED_HOME"),
235  sr_project, /* project name for res file */
236  sr_project, /* project name for res file */
237  sr_project); /* project name for out file */
238 
239  if (system (line_buffer)) {SYS_ERROR_TO_LOG(line_buffer);}
240 
241 /* average IV curves for mirrored and original geometry */
242  sprintf(line_buffer,
243  "\"%s/bin/c_mixtiv\" -t1 \"%s.res_org\" -t2 \"%s.res_mir\" -o \"%s.res\" > \"%s.out\"",
244  getenv("CLEED_HOME"),
245  sr_project, /* project name for res_org file */
246  sr_project, /* project name for res_mir file */
247  sr_project, /* project name for res file */
248  sr_project); /* project name for out file */
249 
250  if (system (line_buffer)) {SYS_ERROR_TO_LOG(line_buffer);}
251 
252 /***********************************************************************
253  Calculate R factor
254 ***********************************************************************/
255 
256 /* changed for Sim. Ann.: range is independent of previous shift. */
257 
258  sprintf(line_buffer,
259  "\"%s\" -t \"%s.res\" -c \"%s.ctr\" -r \"%s\" -s %.2f,%.2f,%.2f> \"%s.dum\"",
260  getenv("CSEARCH_RFAC"), /* R factor program name */
261  sr_project, /* project name for the. file */
262  sr_project, /* project name for control file */
263  RFAC_TYP, /* type of R factor */
264  - RFAC_SHIFT_RANGE, /* initial shift */
265  + RFAC_SHIFT_RANGE, /* final shift */
266  RFAC_SHIFT_STEP, /* step of shift */
267  sr_project); /* project name for output file */
268 
269  #ifdef CONTROL
270  fprintf(STDCTR,"(sr_evalrf %d): calculate R factor:\n %s\n",
271  n_eval, line_buffer);
272  #endif
273 
274  if (system (line_buffer)) {SYS_ERROR_TO_LOG(line_buffer);}
275 
276 /* Read R factor value from output file */
277 
278  sprintf(line_buffer,"%s.dum", sr_project);
279  io_stream = fopen(line_buffer, "r");
280 
281  while( fgets(line_buffer, STRSZ, io_stream) != NULL)
282  {
283  if(
284  #ifdef REAL_IS_DOUBLE
285  (iaux = sscanf(line_buffer, "%lf %lf %lf", &rfac, &faux, &shift) )
286  #endif
287  #ifdef REAL_IS_FLOAT
288  (iaux = sscanf(line_buffer, "%f %f %f", &rfac, &faux, &shift) )
289  #endif
290  == 3) break;
291  }
292 
293 /* Stop with error message if reading error */
294  if( iaux != 3)
295  {
296  log_stream = fopen(log_file, "a");
297  fprintf(log_stream, "***error (sr_eval): failed to read output from '%s'\n",
298  getenv("CSEARCH_RFAC"));
299  fclose(log_stream);
300  exit(1);
301  }
302 
303  fclose (io_stream);
304 
305  #ifdef CONTROL
306  fprintf(STDCTR," rfac = %.4f\n", rfac);
307  #endif
308 
309 /***********************************************************************
310  If this is the minimum R factor copy *.res file to *.rmin
311 ***********************************************************************/
312 
313  if(rfac < rfac_min)
314  {
315 
316  /* remove dependence on 'cp' system call */
317  old_path = (char *) malloc(sizeof(char) * (strlen(sr_project)+5));
318  new_path = (char *) malloc(sizeof(char) * (strlen(sr_project)+5));
319 
320  strcpy(old_path, sr_project);
321  strcpy(new_path, sr_project);
322  strcat(old_path, ".res");
323  strcat(new_path, ".rmin");
324  if (copy_file(old_path, new_path)) {COPY_ERROR_TO_LOG(old_path, new_path);}
325 
326  strcpy(old_path, sr_project);
327  strcpy(new_path, sr_project);
328  strcat(old_path, ".par");
329  strcat(new_path, ".pmin");
330  if (copy_file(old_path, new_path)) {COPY_ERROR_TO_LOG(old_path, new_path);}
331 
332  rfac_min = rfac;
333  }
334 
335  rfac_max = MAX(rfac, rfac_max);
336 #endif /* SHORTCUT */
337 
338 /***********************************************************************
339  Write parameters, R factor and time to *.log file
340 ***********************************************************************/
341 
342  log_stream = fopen(log_file, "a");
343 
344  fprintf(log_stream, "#%3d par:", n_eval);
345  for(i_par=1; i_par <= sr_search->n_par_geo; i_par++)
346  fprintf(log_stream, " %.3f", par[i_par]);
347 
348 
349 /*Inserted a routine to print the angles in the
350  log-file for the angle search:
351 */
352  if(sr_search->sr_angle)
353  {
354  fprintf(log_stream, " theta:%.2f",(par[sr_search->i_par_theta]*FAC_THETA));
355  fprintf(log_stream, " phi:%.2f", (par[sr_search->i_par_phi ]*FAC_PHI) );
356  }
357 
358 
359  fprintf(log_stream," rf:%.4f sh: %.1f rg:%.4f rt:%.4f ",
360  rfac, shift, rgeo, rfac + rgeo);
361 
362  t_time = time(NULL);
363  l_time = localtime(&t_time);
364  fprintf(log_stream," dt: %s", asctime(l_time) );
365 
366  fclose(log_stream);
367 
368 /***********************************************************************
369  Return R factor
370 ***********************************************************************/
371 
372  return (rfac + rgeo);
373 }
