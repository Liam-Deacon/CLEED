27 /************************************************************************
28 
29  DESCRIPTION
30 
31  Create the matrix product of a scattering matrix for a single Bravais
32  layer "i" Tii and propagator matrix to another lattice "j" Gij used in
33  the combined space method to compute the matrix for scattering between
34  two (or more) periodic planes of scatterers.
35 
36 
37  INPUT:
38 
39  mat Gij - (output) multiple scattering matrix in (l,m)-space.
40  mat Llm - (input) lattice sum:
41 
42  Llm = (-1)^m * 4 PI * Ylm (cos(theta) = 0, phi = 0) *
43  * sum(R) [ H(1)l(k*|R|) * exp( i(-kin*R + m*phi(R)) ) ]
44 
45  (in natural order: (l,m) = (0,0); (-1,1); (0,1); (1,1); ...)
46 
47  mat Tii - (input) Scattering matrix for the single periodic layer of type
48  "i" (not used at the moment)
49 
50  Tii is produced by leed_ms_tmat_ii
51 
52  int l_max - max angular momentum quantum number.
53 
54  DESIGN:
55 
56  * General *
57 
58  First create -Gij:
59 
60  Gij(l1,m1; l2,m2) = -S[ L(l3,m3) * C(l3,m3,l1,m1,l2,m2)]
61 
62  L(l3,m3) = lattice sum (created in leed_ms_lsum_ij)
63  C(l3,m3,l1,m1,l2,m2) = C.G. coefficients.
64 
65  Multiply with Tii from l.h.s.
66 
67  FUNCTION CALLS:
68 
69  cg - C.G. coefficients
70  mat* - matrix library
71 
72  RETURN VALUE:
73 
74 Gij
75 
76 NULL if failed (and EXIT_ON_ERROR is not defined)
77 
78 *************************************************************************/
79 {
80 int iaux;
81 int l1,m1, l2,m2;
82 int off_row, off_ij;
83 
84 int l3,m3;
85 int l3_min, l3_max;
86 
87 int i3;
88 
89 real faux_r;
90 real sign;
91 
92 /*************************************************************************
93  Check the input matrices Llm and Tii
94  Make sure that the C.G. coefficients are available
95  (by calling mk_cg_coef)
96 *************************************************************************/
97  if (matcheck(Llm) < 1)
98  {
99 #ifdef ERROR
100  fprintf(STDERR," *** error (leed_ms_tmat_ij): improper input for lattice sum\n");
101 #endif
102 #ifndef EXIT_ON_ERROR
103  return(NULL);
104 #endif
105 #ifdef EXIT_ON_ERROR
106  exit(1);
107 #endif
108  }
109 
110  if (matcheck(Tii) < 1)
111  {
112 #ifdef ERROR
113  fprintf(STDERR,
114  " *** error (leed_ms_tmat_ij): improper input for atomic scattering\n");
115 #endif
116 #ifndef EXIT_ON_ERROR
117  return(NULL);
118 #endif
119 #ifdef EXIT_ON_ERROR
120  exit(1);
121 #endif
122  }
123 
124  if(mk_cg_coef(l_max) != 0)
125  {
126 #ifdef WARNING
127  fprintf(STDWAR,
128  " * warning (leed_ms_tmat_ij): C.G. coefficients had to be recalculated\n");
129 #endif
130  }
131 
132 /* Allocate Gij */
133 
134  iaux = (l_max + 1) * (l_max + 1);
135  Gij = matalloc(Gij, iaux, iaux, NUM_COMPLEX);
136 
137 #ifdef CONTROL
138  fprintf(STDCTR,"\n(leed_ms_tmat_ij): Llm:\n");
139  for(i3 = 1; i3 <= Llm->rows; i3++)
140  printf("(%8.5f, %8.5f)", Llm->rel[i3], Llm->iel[i3]);
141  printf("\n");
142 #endif
143 
144 /*************************************************************************
145  Loop over (l1,m1),(l2,m2): Set up -Gij(l1,m1; l2,m2).
146 
147  -Gij(l1,m1; l2,m2) = -S[ Lij(l3,m3) * C(l1,m1,l2,m2,l3,m3)]
148 
149  L(l3,m3) = lattice sum
150  = -8 PI * i^(l+1) *
151  sum(P) [ Ylm (rj-ri+P) * H(1)l(k*|rj-ri+P|) * exp( i(-kin*P)]
152  C(l3,m3,l1,m1,l2,m2) = C.G. coefficients.
153 *************************************************************************/
154  for(l1 = 0, off_row = 0; l1 <= l_max; l1 ++)
155  {
156  for(m1 = -l1; m1 <= l1; m1 ++, off_row ++ )
157  {
158 /*
159  off_row = l1*(l1+1) - m1;
160 */
161  for(l2 = 0, off_ij = off_row * Gij->cols + 1; l2 <= l_max; l2 ++)
162  {
163  for(m2 = -l2; m2 <= l2; m2 ++, off_ij ++ )
164  {
165  /*
166  Loop over l3 (inner-most loop): calculate the elements of Gij
167 
168  - l3_min: l3 >= |m3|, |l2-l1|.
169  - Only even (l1 + l2 + l3) lead to nonzero C.G.-coefficients
170  => if (l1 + l2 + l3_min) is odd, increment l3_min by 1.
171  => increment l3 by 2 in the inner loop.
172  - The sign (-1)^(m2+1) is due to different definitions of C.G.C's
173  and the "-" in front of the summation.
174  */
175 /*
176  off_ij = off_row * Gij->cols + l2*(l2+1) - m2 + 1;
177 */
178  m3 = m2 - m1;
179 
180  l3_min = MAX(abs(-m3), abs(l2-l1));
181  l3_min += (l1 + l2 + l3_min)%2;
182  l3_max = l2+l1;
183 
184  sign = M1P(m2+1);
185 
186 
187  off_ij = (l1*(l1+1) - m1) * Gij->cols + l2*(l2+1) - m2 + 1;
188  Gij->rel[off_ij] = Gij->iel[off_ij] = 0.;
189 
190  i3 = l3_min*(l3_min + 1) - m3 + 1;
191 
192  for(l3 = l3_min; l3 <= l3_max; l3 += 2 )
193  {
194 
195  faux_r = sign*cg(l3,+m3, l2,+m2, l1,-m1);
196 /*
197  printf("%2d %2d %2d %2d %2d %2d : %.5f\n",
198  l1,m1, l2,m2, l3,m3, faux_r);
199 */
200  Gij->rel[off_ij] += Llm->rel[i3] * faux_r;
201  Gij->iel[off_ij] += Llm->iel[i3] * faux_r;
202 
203  /*
204  l3 is incremented by 2
205  => i3 is incremented by 2*(l3+1) + 2*(l3+2) = 4*l3 + 6
206  sign = -sign;
207  */
208  i3 += 4*l3 + 6;
209  }
210  } /* m2 */
211  } /* l2 */
212  } /* m1 */
213  } /* l1 */
214 
215 #ifdef CONTROL
216  fprintf(STDCTR,"\n(leed_ms_tmat_ij): Gij: \n");
217  matshow (Gij);
218 #endif
219 
220 /*************************************************************************
221  Final matrix multiplication: Gij -> -Tii * Gij
222 *************************************************************************/
223  Gij = matmul(Gij, Tii, Gij);
224 
225 #ifdef CONTROL
226  fprintf(STDCTR,"\n(leed_ms_tmat_ij): Tii*Gij: \n");
227  matshow (Gij);
228 #endif
229 
230  return(Gij);
231 } /* end of function leed_ms_tmat_ij */
