27 /************************************************************************
28 
29  Decide which atoms belong to which layer:
30  - group atoms to composite layers if necessary;
31  - determine inter layer vectors;
32 
33 
34  input parameters:
35 
36  leed_cryst_t *par (input, output)
37  structure that contains all geometrical and nongeometrical
38  parameters of the bulk except atom positions and types.
39  Output will be written to structure element layers.
40

42  structure that contains all geometrical and nongeometrical
43  parameters of the atoms to be grouped (see "leed_def.h").
44  (The list of atoms must be ordered according to their z coordinate,
45  SMALLEST z first.) It will be modified during the execution
46  of this function.
47 
48  return value:
49  number of layers (i_layer) if o.k.
50 
51  DESIGN:
52 
53 
54  NOTE:

57  layer.
58 
59 *************************************************************************/

88  b2[1] = par->b[2]; b2[2] = par->b[4];

92  max. number of layers = number of atoms

97 /************************************************************************
98  i_layer indicates the layer which an atom belongs to; it will eventually
99  be the total number of layers.
100  orig keeps track of the position of the first atom with respect
101  to the overall coordinate system.
102  vaux is used to store the z-position of the topmost atom in the layer.
103 *************************************************************************/

106  no_of_atoms[i_layer] = 1;
107  atom_list[0].layer = i_layer;

110  orig[1] = orig[2] = 0.;
111

116  *(vec + 3*i_layer + i_c) = 0.;
117 
118  atom_list[0].pos[3] = 0.;
119 
120  for(i_atoms=1; i_atoms<n_atoms; i_atoms++)
121  {
122 
123 #ifdef CONTROL

125  atom_list[i_atoms].pos[1]*BOHR, atom_list[i_atoms].pos[2]*BOHR,
126  atom_list[i_atoms].pos[3]*BOHR,
127  R_fabs(atom_list[i_atoms-1].pos[3]+vaux[3]-atom_list[i_atoms].pos[3])*BOHR);
128 #endif

133 /*********************************************************************
134  New layer:
135  - check, whether the previous layer contains only one atom
136  (no_of_atoms[i_layer] == 1).

138  - set up new inter layer vector (vec[3*i_layer + i_c]);
139  - set up new origin of the layer (vaux);
140  - increase i_layer;

145 #endif

148  *(vec + 3*i_layer + 1) = 0.;
149  *(vec + 3*i_layer + 2) = 0.;
150  *(vec + 3*i_layer + 3) =
151  atom_list[i_atoms].pos[3] - atom_list[i_atoms-1].pos[3] - vaux[3];
152

155

161  orig[1] = atom_list[i_atoms-1].pos[1];

164  else

171  *(vec + 3*i_layer + 2) = - (atom_list[i_atoms-1].pos[2]);
172 
173  atom_list[i_atoms-1].pos[1] = 0.;
174  atom_list[i_atoms-1].pos[2] = 0.;
175  }
176

187

193 
194 #ifdef CONTROL_X

199  {

208  *(vec + 3*(i_layer-1) + 2) += atom_list[n_atoms-1].pos[2];
209  }

229 
230 #ifdef CONTROL_X

232  fprintf(STDCTR,"(leed_inp_overlayer): vec_org: %.4f %.4f %.4f \n",
233  *(vec+3*i+1) *BOHR,

236 #endif
