55 int mk_cg_coef(size_t l_max)
56 
57 /************************************************************************
58 
59  DESCRIPTION:
60 
61  Calculate all Clebsh Gordan coefficients
62  C( l1, m1, l2, m2, l3, m3)
63  for 0 <= l1 <= 2*l_max, 0 <= l2,l3 <= l_max.
64 
65  INPUT:
66 
67  int l_max - max angular momentum for output.
68 
69  DESIGN:
70 
71  * Storage Scheme *
72 
73  The storage scheme is:
74 
75  C(l1,m1,l2,m2,l3,m3) -> cg_coef + (l1 * (l1 + 1)/2 + m1pm) * st_fac1 +
76  (l2 * (l2 + 1) + m2pm) * st_fac2 +
77  l3/2;
78 
79  Since C(l1,m1,l2,m2,l3,m3) = C(l1,-m1,l2,-m2,l3,-m3) only the coefficients
80  for positive values of m1 are stored (m2 can be positive of negative).
81  The coefficients of -m1 and m2 are the same as for m1 and -m2.
82 
83  Because of the condition
84  l1 + l2 + l3 = even
85  Only half of the possible values of l3 lead to nonzero C.G.-C's. Only
86  these are stored (offset 0: l3 = 0/1, 1: l3 = 2/3 etc.)
87  Because of
88  m1 = m2 + m3
89  Only one valid value of m3 exists per (m1,m2)-pair.
90 
91  The ranges of l1,l2,l3 are:
92  0 <= l1 <= 2*l_max, 0 <= m1 <= l1;
93  0 <= l2,l3 <= l_max, -l2 <= m2 <= l2;
94 
95  Memory requirements for the static array cg_coef:
96 
97  (2*l_max+1)*(2*l_max+2)/2 * (l_max+1)^2 * (l_max/2+1) * sizeof(double)
98 
99  l_max = 6 : 142,688 bytes
100  l_max = 8 : 495,720 bytes
101  l_max = 10 : 1,341,648 bytes
102  l_max = 12 : 3,075,800 bytes
103  l_max = 14 : 6,264,000 bytes
104  l_max = 16 : 11,673,288 bytes
105 
106  Return values:
107 
108  -1 if an error occured (and if EXIT_ON_ERROR is not defined).
109  0 if C.G. had been calculated before (l_max_coef >= l_max)
110  1 if C.G. were calculated in this call of mk_cg_coef (l_max_coef == l_max)
111  >1 number of warnings accounted.
112 
113 *************************************************************************/
114 {
115 int l1, l2, l3; /* angular momentum quantum numbers */
116 int m1, m2, m3;
117 int m1pm, m2pm, m3pm;
118 int lll;
119 int m2_min, m2_max;
120 
121 int l1_s, l2_s, l3_s; /* permutated indices */
122 int i_st1, i_st2; /* storage address */
123 
124 int i_op, i_warn;
125 int i, i_min, i_max;
126 int iaux;
127 
128 double *fac; /* list of factorials */
129 double fac_m, sum_m; /* factors in the calculation of the C.G.C.'s */
130 double fac_l, fac_ls;
131 double sign;
132 
133 /*****************************************
134  Check if the C.G.C had been
135  calculated already:
136 *****************************************/
137 
138 /*
139  If yes, return 0.
140 */
141  if( (l_max <= l_max_coef) && (cg_coef != NULL) )
142  return(0);
143 
144 /*
145  If not, allocate memory for cg_coef
146 */
147  iaux = (2*l_max + 1)*(2*l_max + 2)/2 * (l_max + 1)*(l_max + 1) * (l_max/2 + 1);
148 
149 #ifdef CONTROL
150  fprintf(STDCTR,"(mk_cg_coef): cg_coef[%d] (%.3f Mb) for l_max = %2d\n",
151  iaux, (real)iaux*sizeof(double) / MBYTE, l_max );
152 #endif
153 
154  if (cg_coef != NULL) free(cg_coef);
155  cg_coef = (double *) calloc (iaux, sizeof(double));
156  if (cg_coef == NULL)
157  {
158 #ifdef ERROR
159  fprintf(STDERR,"(mk_cg_coef): allocation error: cg_coef[%d] = %d bytes\n",
160  iaux, iaux*sizeof(double) );
161 #endif
162 #ifdef EXIT_ON_ERROR
163  exit(1);
164 #endif
165 #ifndef EXIT_ON_ERROR
166  return(-1);
167 #endif
168  }
169 
170 /*
171  Static variables used to retrieve the C.G.C's
172 */
173  st_fac1 = (l_max + 1)*(l_max + 1) * (l_max/2 + 1);
174  st_fac2 = (l_max/2 + 1);
175  l_max_coef = l_max;
176 
177 
178 /*
179  Produce a list of factorials
180 */
181 
182  iaux = 4*l_max + 1;
183  fac = (double *) malloc( (iaux + 1) * sizeof(double) );
184 
185  for (fac[0] = 1. , i = 1; i <= iaux; i ++ )
186  {
187  fac[i] = fac[i-1] * i;
188  }
189 
190 
191 /*****************************************
192  Start calculation of C.G. coefficients:
193  Loops over l2,l3,l1,m1,m2
194 *****************************************/
195 
196  i_op = 0; i_warn = 0;
197  for (l2 = 0; l2 <= l_max; l2 ++)
198  {
199  for (l3 = 0; l3 <= l2; l3 ++)
200  {
201  for (l1 = (l2-l3); l1 <= l2+l3; l1 += 2)
202  /*
203  The sum l1+l2+l3 must be even, i.e. for given l1 and l3, l2 can be
204  incremented by 2. For the start value l2 = |l3-l1| this
205  condition is always fulfilled.
206  */
207  {
208 /*
209  Calculate fac_l
210 */
211  lll = (l1+l2+l3)/2;
212  fac_l = fac[2*(lll - l3)] * fac[lll] /
213  (fac[2*lll + 1] * fac[lll -l1] * fac[lll -l2] * fac[lll -l3]);
214 
215  for (m1pm = 0; m1pm <= l1; m1pm ++)
216  {
217  m2_min = MAX((m1pm-l3),-l2);
218  m2_max = MIN((m1pm+l3), l2);
219 
220  for ( m2pm = m2_min ; m2pm <= m2_max; m2pm ++)
221  /*
222  The condition |m1| >= |m2| >= |m1-m2| is not automatically fulfilled
223  */
224  {
225  m3pm = m1pm-m2pm;
226  m1 = abs(m1pm);
227  m2 = abs(m2pm);
228  m3 = abs(m3pm);
229 
230 #ifdef CONTROL_MK1
231  fprintf(STDCTR,
232  "(mk_cg_coef): > [%1d,%1d(%1d)%1d,%1d(%1d)%1d,%1d(%1d)]\n",
233  l1,m1,m1pm,l2,m2,m2pm,l3,m3,m3pm);
234 #endif
235 
236 /*
237  Find the address in storage space cp_coef:
238  C(l2,m2,l3,m3) and C(l3,m3,l2,m2) are identical:
239 */
240  i_st1 = (l1 * (l1 + 1)/2 + m1pm) * st_fac1 +
241  (l2 * (l2 + 1) + m2pm) * st_fac2 + l3/2;
242  i_st2 = (l1 * (l1 + 1)/2 + m1pm) * st_fac1 +
243  (l3 * (l3 + 1) + m3pm) * st_fac2 + l2/2;
244 
245 /*
246  Check the correct order of m1,m2 and m3. Change the order if necessary.
247  l1_s, l2_s, l3_s are used for a different purpose now !
248 */
249  fac_ls = fac_l;
250  l1_s = l1; l2_s = l2; l3_s = l3;
251 
252  if ((m2 > m1) || (m3 > m1))
253  {
254  if (m3 > m2)
255  {
256  l1_s = l3; m1 = m3;
257  l3_s = l1; m3 = abs(m1pm);
258  l2_s = l2;
259  }
260  else
261  {
262  l1_s = l2; m1 = m2;
263  l2_s = l1; m2 = abs(m1pm);
264  l3_s = l3;
265  }
266 /*
267  Check if l2_s is still bigger than l3_s. If not, exchange l2_s and l3_s.
268  fac_l needs to be recalculated.
269 */
270  if(l2_s < l3_s)
271  {
272  iaux = l2_s; l2_s = l3_s; l3_s = iaux;
273  iaux = m2; m2 = m3; m3 = iaux;
274  }
275 
276  fac_ls = fac[2*(lll - l3_s)] * fac[lll] /
277  (fac[2*lll + 1]*fac[lll -l1_s]*fac[lll -l2_s]*fac[lll -l3_s]);
278  }
279 #ifdef CONTROL_MK1
280  fprintf(STDCTR,
281  "[%1d,%1d,%1d,%1d,%1d,%1d] > [%1d,%1d,%1d,%1d,%1d,%1d]",
282  l1,m1pm,l2,m2pm,l3,m3pm, l1_s,m1,l2_s,m2,l3_s,m3);
283  fprintf(STDCTR," fac_ls = %e\n", fac_ls);
284 #endif
285 
286 /*
287  Calculate fac_m:
288 
289  According to Pendry's book a factor (4*PI)^-1/2 must be multiplied with
290  the plain coefficients.
291 
292  ((4*PI)^-1/2 = .28209478969020929917)
293 */
294  fac_m = fac[l3_s - m3] * fac[l1_s - m1] / fac[l2_s - m2];
295  fac_m *= fac[l2_s + m2] * fac[l3_s + m3] / fac[l1_s + m1];
296  fac_m *= (2*l1_s + 1.) * (2*l2_s + 1.) * (2*l3_s + 1.);
297  fac_m = sqrt(fac_m)*.28209478969020929917;
298 
299 /*
300  Calculate sum_m:
301  i_min, i_max are the boundaries for summation over i.
302 */
303  i_min = MAX(0, (-l2_s + l3_s - m1));
304  i_max = MIN(l1_s - m1, l3_s - m3);
305  i_max = MIN( (l2_s + l3_s - m1), i_max);
306  sign = M1P(i_min);
307 
308  for(i = i_min, sum_m = 0.; i<= i_max; i++, sign = -sign )
309  {
310  sum_m += sign * fac[l1_s + m1 + i] * fac[l2_s + l3_s - m1 - i] /
311  (fac[l1_s - m1 - i] * fac[l2_s - l3_s + m1 + i] *
312  fac[l3_s - m3 - i] * fac[i]);
313  }
314 
315 /*
316  Calculate the C.G. coefficient from the factors
317  fac_ls, fac_m and sum_m.
318 
319  Sign compatible with BLM:
320  sign = M1P(lll - l2_s - m3 + (m1+m2+m3)/2);
321  Sign compatible with CA:
322  sign = M1P(- l2_s - m3 - m3pm + (m1+m2+m3)/2);
323 */
324  sign = M1P(lll - l2_s - m3 + (m1+m2+m3)/2);
325  cg_coef[i_st1] = sign * fac_ls * fac_m * sum_m;
326  cg_coef[i_st2] = sign * fac_ls * fac_m * sum_m;
327 
328 #ifdef WARNING
329  if (cg_coef[i_st1] > WARN_LEVEL)
330  {
331  fprintf(STDWAR,
332  "* warning (mk_cg_coef): CG-C[%2d %2d %2d %2d %2d %2d] = %9.6f\n",
333  l1, m1pm, l2, m2pm, l3, m3pm, cg_coef[i_st1]);
334  i_warn ++;
335  }
336 #endif
337 
338 #ifdef CONTROL_MK
339  fprintf(STDCTR," CG-C[%2d%2d%2d%2d%2d%2d] = %9.6f",
340  l1, m1pm, l2, m2pm, l3, m3pm, cg_coef[i_st1]);
341  fprintf(STDCTR," A*C = %9.6f; B = %9.1f", fac_l * fac_m, sum_m);
342  fprintf(STDCTR," i_st1: %d\n", i_st1);
343 #endif
344  i_op++;
345 
346  } /* m2 */
347  } /* m3 */
348  } /* l3 */
349  } /* l2 */
350  } /* l1 */
351 
352 #ifdef CONTROL
353  fprintf(STDCTR,"(mk_cg_coef): number of operations = %d\n", i_op);
354 #endif
355 #ifdef WARNING
356  if (i_warn)
357  fprintf(STDWAR,"* (mk_cg_coef): number of warnings = %d\n", i_warn);
358 #endif
359 
360  free(fac);
361 
362  return(1+i_warn);
363 } /* end of function mk_cg_coef */

431 double cg (int l1, int m1, int l2, int m2, int l3, int m3)
432 
433 /************************************************************************
434 
435  DESCRIPTION:
436 
437  Returns the Clebsh Gordan coefficient
438  C( l1, m1, l2, m2, l3, m3)
439  for l1 <= 2* l_max_coef, l2/l3 <= l_max_coef
440 
441  INPUT:
442 
443  int l1, l2, l3, m1, m2, m3 - angular momentum quantum numbers
444 
445  DESIGN:
446 
447  Reverse operation to the storage scheme used in mk_cg_coef.
448 
449  The function does not check if the quantum numbers are within the limits
450  given by l_max_coef. This has to be checked outside the function using
451  cg_info.
452 
453 *************************************************************************/
454 {
455 int i_st;
456 
457 
458 /*
459  First: trivial results:
460  l1 + l2 + l3 not even
461  m1 != m2 + m3
462 */
463 
464  if ( ODD(l1 + l2 + l3) ) return (0.);
465  if ( m1 != m2 + m3) return (0.);
466 
467 /*
468  Find the address in storage space cp_coef:
469 
470  Only C.G.C's for m1 >= 0 are stored. If m1 < 0, look for
471  C.G.C.[-m1,-m2] = C.G.C.[m1,m2]
472 */
473 
474  if (m1 < 0) { m1 = -m1; m2 = -m2; }
475 
476 #ifdef CONTROL_CG
477  fprintf(STDCTR,"(cg:) l1:%2d, m1:%2d, l2:%2d, m2:%2d, l3:%2d, m3:%2d\n",
478  l1,m1,l2,m2,l3,m3 );
479 #endif
480 
481  i_st = (l1 * (l1 + 1)/2 + m1) * st_fac1 +
482  (l2 * (l2 + 1) + m2) * st_fac2 + l3/2;
483 
484  return(cg_coef[i_st]);
485 
486 
487 } /* end of function cg */
