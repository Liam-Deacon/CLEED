17 int patt_draw_cairo(const patt_drawing *drawing)
18 {
19  cairo_surface_t *surface;
20  cairo_t *cr;
21  size_t i_file, i_spot, i_domain;
22  patt_color_rgb color = PATT_BLACK;
23  pattern *pat = NULL;
24  FILE *f = NULL;
25 
26  char *copyright = (char*) malloc((BUFSIZ)*sizeof(char));
27  char *tok = NULL;
28  char str_buf[BUFSIZ];
29 
30  strcpy(copyright, PROG_COPYRIGHT);
31  tok = strtok(copyright, "\n");
32 
33  /* create surface */
34  switch (drawing->format)
35  {
36  case PATT_PNG:
37  surface = cairo_image_surface_create(CAIRO_FORMAT_ARGB32,
38  PATT_PAGE_WIDTH, PATT_PAGE_HEIGHT);
39  break;
40 
41  case PATT_PDF:
42  surface = cairo_pdf_surface_create(drawing->output_filename,
43  PATT_PAGE_WIDTH, PATT_PAGE_HEIGHT);
44  break;
45 
46  case PATT_SVG:
47  surface = cairo_svg_surface_create(drawing->output_filename,
48  PATT_PAGE_WIDTH, PATT_PAGE_HEIGHT);
49  break;
50 
51  case PATT_EPS: case PATT_PS: default:
52  surface = cairo_ps_surface_create(drawing->output_filename,
53  PATT_PAGE_WIDTH, PATT_PAGE_HEIGHT);
54  if (drawing->format == PATT_EPS) cairo_ps_surface_set_eps(surface, true);
55 
58  /* title */
59  sprintf(str_buf, "%%Title: %s", drawing->title.label);
60  cairo_ps_surface_dsc_comment(surface, str_buf);
61 
62  /* copyright */
63  while(tok)
64  {
65  sprintf(str_buf, "%%Copyright: %s", tok);
66  tok = strtok(NULL, "\n");
67  cairo_ps_surface_dsc_comment(surface, str_buf);
68  }
69 
70  /* program info */
71  sprintf(str_buf, "%%Creator: %s - Version %s", PROG, PROG_VERSION);
72  cairo_ps_surface_dsc_comment(surface, str_buf);
73 
74  break;
75  }
76  cr = cairo_create(surface);
77 
78  /* set origin to center of image */
79  cairo_translate(cr, PATT_PAGE_WIDTH, PATT_PAGE_HEIGHT);
80 
81  /* set options */
82  cairo_select_font_face(cr, "Serif", CAIRO_FONT_SLANT_NORMAL,
83  CAIRO_FONT_WEIGHT_NORMAL);
84 
85  cairo_set_font_size(cr, 13);
86 
87 
88  /***************************** STRUCTURE LOOP ****************************/
89  for (i_file = 0; i_file < drawing->n_files; i_file++)
90  {
91  if (strlen(drawing->input_files[i_file]))
92  {
93  if ((f = fopen(drawing->input_files[i_file], "r")) != NULL)
94  {
95  pat = pattern_read(f);
96  }
97  else
98  {
99  fprintf(stderr, "***error (patt_draw_cairo): "
100  "cannot open '%s'\n", drawing->input_files[i_file]);
101  exit(PATT_READ_ERROR);
102  }
103  }
104  /* draw substrate structure */
105  spots *gs_spots = pattern_calculate_substrate_spots(pat);
106 
107  switch (drawing->color_scheme)
108  {
109  case PATT_GRAYSCALE_SCHEME:
110  color = grays[i_file % NUM_GRAYS];
111  break;
112  case PATT_COLOR_SCHEME:
113  patt_color_copy(&color, patt_color_from_name(
114  colors[i_file % NUM_COLORS][SPOT_GS]));
115  break;
116  case PATT_MONOCHROME_SCHEME: default:
117  color = PATT_BLACK;
118  break;
119  }
120 
121  /* set spot properties */
122  patt_color_copy(&gs_spots->color, &color);
123  cairo_set_source_rgb(cr, color.red, color.green, color.blue);
124  cairo_set_line_width(cr, gs_spots->stroke_width);
125 
126  for (i_spot = 0; i_spot < gs_spots->n_spots; i_spot++)
127  {
131  patt_draw_cairo_spot(cr, &gs_spots->spots[i_spot],
132  gs_spots->radius, spots_get_shape(gs_spots));
133  if (gs_spots->fill == true) cairo_fill(cr);
134  if (gs_spots->indexing == true)
135  {
136  patt_draw_cairo_label(cr, &gs_spots->spots[i_spot]);
137  }
138 
139  }
140 
141  /* draw superstructure spots */
142  for (i_domain = 0; i_domain < pat->n_domains; i_domain++)
143  {
144  spots *ss_spots = pattern_calculate_superstructure_spots(pat, i_domain);
145 
146  cairo_set_line_width(cr, ss_spots->stroke_width);
147  cairo_set_font_size(cr, abs(ss_spots->font_size));
148 
149  patt_color_copy(&ss_spots->color, &gs_spots->color);
150  spots_set_shape(ss_spots, i_domain);
151  ss_spots->font_size = 12;
152  ss_spots->radius = RADIUS_SS;
153 
154  for (i_spot = 0; i_spot < ss_spots->n_spots; i_spot++)
155  {
157  patt_draw_cairo_spot(cr, &ss_spots->spots[i_spot],
158  ss_spots->radius, spots_get_shape(ss_spots));
159  if (gs_spots->fill == true) cairo_fill(cr);
160  if (gs_spots->indexing == true)
161  {
162  patt_draw_cairo_label(cr, &ss_spots->spots[i_spot]);
163  }
164  }
165 
166  spots_free(ss_spots);
167  }
168 
169  /* clean up */
170  spots_free(gs_spots);
171  pattern_free(pat);
172  fclose(f);
173 
174  } /* for i_file */
175 
178  /* eV */
179  if (drawing->eV.visible)
180  {
181  patt_draw_cairo_text(cr, &drawing->eV, CAIRO_FONT_SLANT_NORMAL,
182  CAIRO_FONT_WEIGHT_NORMAL);
183  }
184 
185  /* screen */
186  if (drawing->screen.visible) patt_draw_cairo_screen(cr, &drawing->screen);
187 
188  /* electron gun */
189  if (drawing->gun.visible) patt_draw_cairo_gun(cr, &drawing->gun);
190 
191  if (drawing->footnote.visible)
192  {
193  patt_draw_cairo_text(cr, &drawing->footnote, CAIRO_FONT_SLANT_NORMAL,
194  CAIRO_FONT_WEIGHT_NORMAL);
195  }
196 
197  if (drawing->title.visible)
198  {
199  patt_draw_cairo_text(cr, &drawing->title, CAIRO_FONT_SLANT_NORMAL,
200  CAIRO_FONT_WEIGHT_BOLD);
201  }
202 
203 
204 
206  switch (drawing->format)
207  {
208  case PATT_PNG:
209  {
210  cairo_surface_write_to_png(surface, drawing->output_filename);
211  break;
212  }
213 
214  case PATT_SVG:
215  case PATT_PDF:
216  case PATT_EPS:
217  case PATT_PS:
218  default:
219  {
220  cairo_show_page(cr);
221  cairo_surface_flush(surface);
222  break;
223  }
224 
225  }
226 
227  /* clean up */
228  cairo_destroy(cr);
229  cairo_surface_finish(surface);
230  cairo_surface_destroy(surface);
231 
232  free(tok);
233  free(copyright);
234 
235  return(PATT_SUCCESS);
236 }

238 void patt_draw_cairo_spot(cairo_t *cr, spot *spot,
239  double spot_size, patt_shape shape)
240 /*
241  draw spots in different shapes:
242  0: circle
243  1: square
244  2: triangle
245 */
246 
247 {
248  double x = spot->x, y = spot->y;
249  int delta;
250 
251  switch(shape)
252  {
253  case(PATT_SQUARE):
254  {
255  /* square */
256  spot_size /= 1.4;
257  cairo_move_to(cr, x, y);
258  cairo_rectangle(cr, x - spot_size, y - spot_size,
259  x + spot_size, y + spot_size);
260  break;
261  }
262  case(PATT_TRIANGLE_UP):
263  {
264  /* up triangle */
265  delta = spot_size * 0.866;
266  cairo_new_path(cr);
267  cairo_move_to(cr, x, y + spot_size);
268  cairo_line_to(cr, x - delta, y - (0.5* spot_size));
269  cairo_line_to(cr, x + spot_size, y + spot_size);
270  cairo_line_to(cr, x + delta, y - (0.5* spot_size));
271  cairo_close_path(cr);
272  break;
273  }
274  case(PATT_TRIANGLE_DOWN):
275  {
276  /* down triangle */
277  delta = spot_size * 0.866;
278  cairo_new_path(cr);
279  cairo_move_to(cr, x, y - spot_size);
280  cairo_line_to(cr, x - delta, y + (0.5* spot_size));
281  cairo_line_to(cr, x + delta, y + (0.5 * spot_size));
282  cairo_close_path(cr);
283  break;
284  }
285  case(PATT_DIAMOND):
286  {
287  /* diamond */
288  cairo_new_path(cr);
289  cairo_move_to(cr, x - spot_size, y);
290  cairo_line_to(cr, x, y + spot_size);
291  cairo_line_to(cr, x + spot_size, y);
292  cairo_line_to(cr, x, y - spot_size);
293  cairo_close_path(cr);
294  break;
295  }
296  case(PATT_HEXAGON):
297  {
298  /* hexagon */
299  delta = spot_size * 0.866;
300  cairo_new_path(cr);
301  cairo_move_to(cr, x, y - spot_size);
302  cairo_line_to(cr, x - delta, y - (0.5*spot_size));
303  cairo_line_to(cr, x - delta, y + (0.5*spot_size));
304  cairo_line_to(cr, x, y + spot_size);
305  cairo_line_to(cr, x + delta, y + (0.5*spot_size));
306  cairo_line_to(cr, x + delta, y - (0.5*spot_size));
307  cairo_close_path(cr);
308  break;
309  }
310  case(PATT_CIRCLE): default:
311  {
312  cairo_arc(cr, x, y, spot_size, 0, 2 * M_PI);
313  break;
314  }
315  } /* switch */
316 }

327 void patt_draw_cairo_arrow(cairo_t *cr, double x1, double y1, double x2,
328  double y2, double stroke_width, double head_size)
329 {
330  double angle = atan2 (y2 - y1, x2 - x1) + M_PI;
331  double arrow_length = sqrt((x2-x1)*(x2-x1) + (y2-y1)*(y2-y1));
332  double head_x1 = x2 + arrow_length * cos(angle - ARROW_ANGLE);
333  double head_y1 = y2 + arrow_length * sin(angle - ARROW_ANGLE);
334  double head_x2 = x2 + arrow_length * cos(angle + ARROW_ANGLE);
335  double head_y2 = y2 + arrow_length * sin(angle + ARROW_ANGLE);
336 
337  double original_width = cairo_get_line_width(cr);
338 
339  cairo_set_line_width(cr, stroke_width);
340  cairo_move_to(cr, x1, y1);
341  cairo_line_to(cr, x2, y2);
342  cairo_line_to(cr, head_x1, head_y1);
343  cairo_line_to(cr, head_x2, head_y2);
344  cairo_close_path(cr);
345 
346  //cairo_set_source_rgb(cr, );
347  cairo_stroke_preserve(cr);
348 
349  //cairo_set_source_rgb(cr, );
350  cairo_fill(cr);
351 
352  /* restore */
353  cairo_set_line_width(cr, original_width);
354 
355 }

357 void patt_draw_cairo_gun(cairo_t *cr, const patt_gun *gun)
358 {
359  patt_color_rgb color = gun->color;
360 
361  cairo_set_source_rgba(cr, color.red, color.blue, color.green, gun->alpha);
362  cairo_set_line_width(cr, gun->stroke_width);
363  cairo_move_to(cr, gun->stroke_width, -gun->stroke_width);
364  cairo_line_to(cr, -(1.005*MAX_RADIUS*cos(3.14182*(gun->angle+0.035))),
365  -(1.005*MAX_RADIUS*sin(3.14182*(gun->angle+0.035))));
366  cairo_line_to(cr, -(1.005*MAX_RADIUS*cos(3.14182*(gun->angle-0.035))),
367  -(1.005*MAX_RADIUS*sin(3.14182*(gun->angle-0.035))));
368  cairo_line_to(cr, -gun->stroke_width, gun->stroke_width);
369  cairo_close_path(cr);
370  cairo_set_source_rgba(cr, color.red, color.blue, color.green, gun->alpha);
371  cairo_fill(cr);
372  cairo_set_source_rgba(cr, color.red, color.blue, color.green, gun->alpha);
373  cairo_arc(cr, 0., 0., gun->radius, 0., 2*M_PI);
374  cairo_stroke(cr);
375  cairo_fill(cr);
376 }

378 void patt_draw_cairo_screen(cairo_t *cr, const patt_screen *screen)
379 {
380  cairo_move_to(cr, 0., 0.);
381  cairo_set_line_width(cr, screen->stroke_width);
382  cairo_set_source_rgba(cr, screen->stroke_color.red, screen->stroke_color.blue,
383  screen->stroke_color.green, screen->alpha);
384  cairo_arc(cr, 0., 0., MAX_RADIUS+(screen->stroke_width*0.5), 0., 2*M_PI);
385  cairo_stroke(cr);
386 
387  if(screen->fill)
388  {
389  cairo_arc(cr, 0., 0., MAX_RADIUS+(screen->stroke_width*0.5), 0., 2*M_PI);
390  cairo_set_source_rgba(cr, screen->fill_color.red, screen->fill_color.blue,
391  screen->fill_color.green, screen->alpha);
392  cairo_stroke(cr);
393  cairo_fill(cr);
394  }
395 }
