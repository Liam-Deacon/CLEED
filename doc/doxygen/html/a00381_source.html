37 void sr_sx_gsl(size_t ndim, real dpos, char *bak_file, char *log_file)
38 
39 /***********************************************************************
40  SIMPLEX METHOD
41 ***********************************************************************/
42 {
43 
44 size_t i_par, j_par;
45 size_t mpar, nfunc=0;
46 real faux;
47 
48 gsl_vector *x = gsl_vector_alloc(ndim);
49 gsl_vector *y = gsl_vector_alloc(ndim);
50 gsl_matrix *p = gsl_matrix_alloc(ndim, ndim);
51 
52 FILE *log_stream;
53 
54 /***********************************************************************
55  SIMPLEX METHOD
56 ***********************************************************************/
57 
58  if( (log_stream = fopen(log_file, "a")) == NULL) { OPEN_ERROR(log_file); }
59  fprintf(log_stream,"=> SIMPLEX SEARCH:\n\n");
60 
61 /***********************************************************************
62  Set up vertex if no vertex file was specified, read vertex otherwise.
63 ***********************************************************************/
64 
65  mpar = ndim + 1;
66 
67  if(strncmp(bak_file, "---", 3) == 0)
68  {
69  fprintf(log_stream,"=> Set up vertex:\n");
70  fclose(log_stream);
71 
72  /* set initial matrix parameters to zero */
73  for(i_par = 1; i_par <= ndim; i_par ++)
74  {
75  gsl_matrix_set(p, 0, i_par-1, 0.);
76  }
77 
78  for (i_par = 1; i_par <= mpar; i_par ++)
79  {
80  for (j_par = 1; j_par <= ndim; j_par ++)
81  {
82  faux = gsl_matrix_get(p, 0, j_par-1);
83  if(i_par == (j_par+1))
84  faux += dpos;
85  gsl_vector_set(x, j_par-1, faux);
86  gsl_matrix_set(p, i_par-1, j_par-1, faux);
87  }
88 
89  #ifdef CONTROL
90  fprintf(STDCTR,"(sr_sx): Calculate function for vertex(%d)\n", i_par);
91  #endif
92 
93  gsl_vector_set(y, i_par-1, sr_evalrf_gsl(x));
94  }
95  }
96  else
97  {
98  fprintf(log_stream, "=> Read vertex from \"%s\":\n", bak_file);
99  sr_rdver_gsl(bak_file, y, p);
100  fclose(log_stream);
101  }
102 
103 /***********************************************************************
104  Enter amoeba
105 ***********************************************************************/
106 
107  #ifdef CONTROL
108  fprintf(STDCTR, "(sr_sx_gsl): Enter amoeba\n");
109  #endif
110 
111  if( (log_stream = fopen(log_file, "a")) == NULL) { OPEN_ERROR(log_file); }
112  fprintf(log_stream, "=> Start search (abs. tolerance = %.3e)\n", R_TOLERANCE);
113  fclose(log_stream);
114 
115  sr_amoeba_gsl(p, y, R_TOLERANCE, &nfunc);
116 
117 /***********************************************************************
118  Write final results to log file
119 ***********************************************************************/
120 
121  #ifdef CONTROL
122  fprintf(STDCTR, "(sr_sx_gsl): %d function evaluations in sr_amoeba_gsl\n",
123  nfunc);
124  #endif
125 
126  if( (log_stream = fopen(log_file, "a")) == NULL) { OPEN_ERROR(log_file); }
127 
128  fprintf(log_stream, "\n=> No. of function evaluations in sr_amoeba_gsl: %3d\n",
129  nfunc);
130  fprintf(log_stream, "=> Vertices and function values of final 3-d simplex:\n");
131 
132 /* print 1st line */
133 
134  fprintf(log_stream, "%3d:",1);
135  for (j_par = 0; j_par < ndim; j_par++ )
136  fprintf(log_stream, "%7.4f ", gsl_matrix_get(p, 0, j_par));
137  fprintf(log_stream,"%7.4f\n", gsl_vector_get(y, 0));
138 
139 /* print other lines */
140  for (i_par = 0; i_par < mpar; i_par++ )
141  {
142  fprintf(log_stream, "%3d ", i_par+1);
143  for (j_par = 0; j_par < ndim; j_par++ )
144  {
145  faux = gsl_matrix_get(p, i_par, j_par);
146  fprintf(log_stream, "%7.4f ", faux);
147  gsl_matrix_set(p, 0, j_par, gsl_matrix_get(p, 0, j_par)+faux);
148  }
149  fprintf(log_stream, "%7.4f\n", gsl_vector_get(y, i_par));
150  faux = gsl_vector_get(y, 0);
151  gsl_vector_set(y, 0, faux+gsl_vector_get(y, i_par) );
152  }
153 
154  fprintf(log_stream, "\navg:");
155  for (j_par=0; j_par < ndim; j_par++)
156  fprintf(log_stream, "%7.4f ", gsl_matrix_get(p, 0, j_par)/mpar);
157  fprintf(log_stream, "%7.4f\n", gsl_vector_get(y, 0)/mpar);
158 
159  fclose(log_stream);
160 
161  /* free memory */
162  gsl_matrix_free(p);
163  gsl_vector_free(x);
164  gsl_vector_free(y);
165 
166 } /* end of function sr_sx */
