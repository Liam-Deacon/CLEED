77  gsl_multimin_function func;

81

84 
85  /* Set all step sizes to 1 */
86  gsl_vector_set_all(ss, 1.0);
87 
88  /* Starting point */
89  x = gsl_vector_alloc(np);
90

92  func.f = &sr_amoeba_eval;
93  func.n = np;

96 
97  s = gsl_multimin_fminimizer_alloc(gsl_multimin_fminimizer_nmsimplex, np);
98  gsl_multimin_fminimizer_set(s, &func, x, ss);
99 
100  /* iterate through minimiser */
101  do {
102  *iter += 1;
103  status = gsl_multimin_fminimizer_iterate(s);
104 
105  if (status != GSL_CONTINUE && status != GSL_SUCCESS)
106  {
107  fprintf(STDERR, "***error (sr_amoeba_gsl): %s\n",
108  gsl_strerror(status));
109  exit(status);
110  break;
111  }
112  size = gsl_multimin_fminimizer_size(s);
113  status = gsl_multimin_test_size(size, epsilon);
114 
115  for(j = 0; j < np; j++)
116  gsl_matrix_set(p, *iter, j, gsl_vector_get(s->x, j));
117 
118  } while (status == GSL_CONTINUE && *iter < MAX_ITER_AMOEBA);

138

142  for (i = 0; i < np+1; i++)
143  {
