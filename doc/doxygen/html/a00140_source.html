19 mat ms_partinv ( mat Minv, mat Mbg, int first_atoms, int l_max)
20 
21 /************************************************************************
22 
23  DESCRIPTION:
24 
25  Invert giant scattering matrix by partitioning.
26 
27  INPUT:
28 
29  mat Minv - output: inverse of input matrix
30  mat Mbg - input: giant scattering matrix (dimension N * (l_max +1)^2)
31 
32  int first_atoms - number of atoms which are in the same plane.
33  int l_max - max angular momentum quantum number.
34 
35  DESIGN:
36 
37  * General *
38 
39  The function follows the formulae given in Num. Rec. (2.7.22 ff/p. 77).
40  The inverse of the matrix
41 
42  Mbg = (UL UR)
43  (LL LR)
44  Can be written as
45  Minv = (P Q)
46  (R S)
47  where the submatrices P, Q, R, and S have the same dimensions as UL, UR,
48  LL, and LR, respectively. They can be calculated as follows:
49 
50  S = [LR - LL*(UL^-1)*UR ]^-1
51  R = -S* LL*(UL^-1)
52  Q = -(UL^-1)*UR * S
53  P = (UL^-1) + (UL^-1)*UR * S * LL*(UL^-1)
54 
55  If the first (first_atoms) atoms are in the same plane parallel to the
56  surface, the upper left submatrix UL (dimension: first_atoms *(l_max +1)^2)
57  can be blockdiagonalized into two blocks with even (l1+m1), (l2+m2)
58  (dimension: first_atoms * (l_max + 1)*(l_max + 2)/2) and odd (l1+m1),
59  (l2+m2) (dimension: first_atoms * (l_max +1 ) * l_max/2).
60  Thus, UL can be inverted by inverting the two blocks which requires less
61  cpu time. There are no savings in the other matrix operations of the above
62  formulae, therefore this method saves more time the more atoms are in the
63  same plane.
64 
65  * Special *
66 
67  First UL is decomposed into two matrices with even (l1+m1), (l2+m2), Maux_a
68  and odd (l1+m1), (l2+m2), Maux_b. Maux_a and Maux_b are inverted separately
69  and reinserted into UL which has now become (UL^-1).
70 
71  If first_atoms *(l_max +1)^2 is equal to the dimensions of Mbg (i.e. all
72  atoms are in the same plane) (UL^-1) is identical to the inverse of Mbg and
73  the function will return (UL^-1). If {first_atoms *(l_max +1)^2} is smaller
74  than the dimensions of Mbg (i.e. not all atoms are in the same plane) S, R,
75  Q, and P are calculated according to the above formulae, whereby
76  Maux_a = LL*(UL^-1) and Maux_b = (UL^-1)*UR are stored as intermediate
77  results in order to be reused.
78 
79  FUNCTION CALLS:
80 
81  matalloc
82  matfree
83  matcheck
84 
85  matext
86  matins
87 
88  matinv
89  matmul
90 
91 
92  RETURN VALUE:
93 
94  Minv, if o.k.
95 
96 *************************************************************************/
97 {
98 int iaux;
99 
100 int i_atoms_1, i_atoms_2;
101 int l1,m1, l2,m2;
102 
103 int iev2;
104 int iod2;
105 int odd1;
106 
107 real *ptr_1, *ptr_2, *ptr_end;
108 
109 mat Maux_a, Maux_b;
110 mat UL, UR, LL, LR;
111 
112 
113  UL = UR = LL = LR = NULL;
114  Maux_a = Maux_b = NULL;
115 
116 /*************************************************************************
117  Extract UL (upper left) from Mbg.
118  Allocate Maux_a, Maux_b.
119 *************************************************************************/
120 
121  iaux = first_atoms * (l_max + 1)*(l_max + 1);
122  UL = matext(UL, Mbg, 1, iaux, 1, iaux);
123 
124  iaux = first_atoms * (l_max + 1)*(l_max + 2)/2;
125  Maux_a = matalloc( Maux_a, iaux, iaux, NUM_COMPLEX);
126  iaux = first_atoms * (l_max +1 ) * l_max/2;
127  Maux_b = matalloc( Maux_b, iaux, iaux, NUM_COMPLEX);
128 
129 /*************************************************************************
130  Loop over (l1,m1),(l2,m2): Set up Maux_a and Maux_b.
131 
132  - UL separates into two blocks with even (l1+m1), (l2+m2) and odd
133  (l1+m1), (l2+m2) which are stored in Maux_a and Maux_b, respectively.
134 *************************************************************************/
135  for(i_atoms_1 = 0, iev2 = 1, iod2 = 1, ptr_1 = UL->rel+1, ptr_2 = UL->iel+1;
136  i_atoms_1 < first_atoms; i_atoms_1 ++)
137  {
138  for(l1 = 0; l1 <= l_max; l1 ++)
139  {
140  for(m1 = -l1; m1 <= l1; m1 ++)
141  {
142  odd1 = ODD(l1+m1);
143  for(i_atoms_2 = 0; i_atoms_2 < first_atoms; i_atoms_2 ++)
144  {
145  for(l2 = 0; l2 <= l_max; l2 ++)
146  {
147  for(m2 = -l2; m2 <= l2; m2 ++, ptr_1 ++, ptr_2 ++)
148  {
149  if(ODD(l2+m2))
150  {
151  if( odd1)
152  {
153  Maux_b->rel[iod2] = *ptr_1;
154  Maux_b->iel[iod2] = *ptr_2;
155  iod2++;
156  }
157  }
158  else
159  {
160  if(!odd1)
161  {
162  Maux_a->rel[iev2] = *ptr_1;
163  Maux_a->iel[iev2] = *ptr_2;
164  iev2++;
165  }
166  } /* else */
167  } /* m2 */
168  } /* l2 */
169  } /* i_atoms_2 */
170  } /* m1 */
171  } /* l1 */
172  } /* i_atoms_1 */
173 
174 #ifdef CONTROL_X
175  fprintf(STDCTR,"(ms_partinv): UL, Maux_a/od: \n");
176 #endif
177 
178 /*************************************************************************
179  Matrix inversion: (Maux_a/od)^-1
180 *************************************************************************/
181 
182  Maux_a = matinv(Maux_a, Maux_a);
183  Maux_b = matinv(Maux_b, Maux_b);
184 
185 #ifdef CONTROL_X
186  fprintf(STDCTR,"(ms_partinv): (Maux_b)^-1: \n");
187 #endif
188 
189 /*************************************************************************
190  Copy (Maux_a)^-1 and (Maux_b)^-1 back into UL in the natural order.
191 *************************************************************************/
192 
193  for(i_atoms_1 = 0, iev2 = 1, iod2 = 1, ptr_1 = UL->rel+1, ptr_2 = UL->iel+1;
194  i_atoms_1 < first_atoms; i_atoms_1 ++)
195  {
196  for(l1 = 0; l1 <= l_max; l1 ++)
197  {
198  for(m1 = -l1; m1 <= l1; m1 ++)
199  {
200  odd1 = ODD(l1+m1);
201  for(i_atoms_2 = 0; i_atoms_2 < first_atoms; i_atoms_2 ++)
202  {
203  for(l2 = 0; l2 <= l_max; l2 ++)
204  {
205  for(m2 = -l2; m2 <= l2; m2 ++, ptr_1 ++, ptr_2 ++)
206  {
207  if(ODD(l2+m2))
208  {
209  if( odd1)
210  {
211  *ptr_1 = Maux_b->rel[iod2];
212  *ptr_2 = Maux_b->iel[iod2];
213  iod2++;
214  }
215  else
216  {
217  *ptr_1 = 0.;
218  *ptr_2 = 0.;
219  }
220  }
221  else
222  {
223  if(!odd1)
224  {
225  *ptr_1 = Maux_a->rel[iev2];
226  *ptr_2 = Maux_a->iel[iev2];
227  iev2++;
228  }
229  else
230  {
231  *ptr_1 = 0.;
232  *ptr_2 = 0.;
233  }
234  } /* else */
235  } /* m2 */
236  } /* l2 */
237  } /* i_atoms_2 */
238  } /* m1 */
239  } /* l1 */
240  } /* i_atoms_1 */
241 
242 /*************************************************************************
243  If all the atoms are in the same plane (i.e. first_atoms * (l_max + 1)^2
244  is equal to the dimension of Mbg) the inversion is already complete.
245 => Return here.
246 *************************************************************************/
247 
248  iaux = first_atoms * (l_max + 1)*(l_max + 1);
249  if(iaux == Mbg->cols)
250  {
251 #ifdef CONTROL
252  fprintf(STDCTR,"(ms_partinv): All atoms are in the same plane.\n");
253 #endif
254 
255  matfree(Maux_a);
256  matfree(Maux_b);
257 
258  Minv = matcop(Minv, UL);
259  matfree(UL);
260 
261  return(Minv);
262  }

264 /*************************************************************************
265  Calculate submatrices of the inverse of Mbg according to:
266 
267  Mbg = (UL UR) Minv = (Mbg)^-1 = (P Q)
268  (LL LR) (R S)
269 
270  where:
271  S = (LR - (LL*UL^-1)*UR )^-1
272  R = -S*(LL*UL^-1)
273  Q = -(UL^-1)*UR * S
274  P = (UL^-1) + (UL^-1)*UR * S * (LL*UL^-1)
275 
276 *************************************************************************/
277 
278 /*************************************************************************
279  First
280  Extract UR (upper right), LL (lower left), and LR (lower right) from Mbg.
281  Allocate Maux_a, Maux_b.
282 *************************************************************************/
283 
284  iaux = first_atoms * (l_max + 1)*(l_max + 1);
285  UR = matext(UR, Mbg, 1, iaux, iaux+1, Mbg->cols);
286  LL = matext(LL, Mbg, iaux+1, Mbg->rows, 1, iaux);
287  LR = matext(LR, Mbg, iaux+1, Mbg->rows, iaux+1, Mbg->cols);
288 
289 #ifdef CONTROL
290  fprintf(STDCTR,"\n(ms_partinv):\tUL(%d x %d) UR(%d x %d)\n",
291  UL->rows,UL->cols, UR->rows,UR->cols);
292  fprintf(STDCTR,"\t\tLL(%d x %d) LR(%d x %d)\n",
293  LL->rows,LL->cols, LR->rows,LR->cols);
294 #endif
295 
296 /*
297  Maux_a = (LL*UL^-1)
298  Maux_b = (LL*UL^-1)*UR ) = Maux_a * UR
299 */
300 
301 #ifdef CONTROL_X
302  fprintf(STDCTR,"(ms_partinv): Maux_a\n");
303 #endif
304 
305  Maux_a = matmul(Maux_a, LL, UL);
306 #ifdef CONTROL_X
307  fprintf(STDCTR,"(ms_partinv): Maux_b\n");
308 #endif
309 
310  Maux_b = matmul(Maux_b, Maux_a, UR);
311 
312 /*
313  Maux_b = -(LR - (LL*UL^-1)*UR) = Maux_b - LR (first real then imag. part)
314  LR = Maux_b^-1 = -S
315 */
316 
317 #ifdef CONTROL_X
318  fprintf(STDCTR,"(ms_partinv): LR\n");
319 #endif
320 
321  iaux = LR->cols * LR->rows;
322  for(ptr_1 = LR->rel+1, ptr_2 = Maux_b->rel+1, ptr_end = LR->rel+iaux;
323  ptr_1 <= ptr_end; ptr_1 ++, ptr_2 ++)
324  { *ptr_2 -= *ptr_1; }
325 
326  for(ptr_1 = LR->iel+1, ptr_2 = Maux_b->iel+1, ptr_end = LR->iel+iaux;
327  ptr_1 <= ptr_end; ptr_1 ++, ptr_2 ++)
328  { *ptr_2 -= *ptr_1; }
329 
330  LR = matinv(LR, Maux_b);
331 
332 /*
333  Maux_b = (UL^-1)*UR
334  R -> LL = - S * (LL*UL^-1) = LR * Maux_a
335  Q -> UR = - (UL^-1)*UR * S = Maux_b * LR
336 */
337 
338 #ifdef CONTROL_X
339  fprintf(STDCTR,"(ms_partinv): LL, UR\n");
340 #endif
341 
342  Maux_b = matmul(Maux_b, UL, UR);
343 
344  LL = matmul(LL, LR, Maux_a);
345  UR = matmul(UR, Maux_b, LR);
346 
347 /*
348  P -> UL = (UL^-1) + (UL^-1)*UR * S * (LL*UL^-1)
349  = UL - Maux_b * LL
350 */
351 
352 #ifdef CONTROL_X
353  fprintf(STDCTR,"(ms_partinv): UL\n");
354 #endif
355 
356  Maux_b = matmul(Maux_b, Maux_b, LL);
357 
358  iaux = UL->cols * UL->rows;
359  for(ptr_1 = UL->rel+1, ptr_2 = Maux_b->rel+1, ptr_end = UL->rel+iaux;
360  ptr_1 <= ptr_end; ptr_1 ++, ptr_2 ++)
361  { *ptr_1 -= *ptr_2; }
362 
363  for(ptr_1 = UL->iel+1, ptr_2 = Maux_b->iel+1, ptr_end = UL->iel+iaux;
364  ptr_1 <= ptr_end; ptr_1 ++, ptr_2 ++)
365  { *ptr_1 -= *ptr_2; }
366 
367 /*
368  S -> LR = -LR
369 */
370 #ifdef CONTROL_X
371  fprintf(STDCTR,"(ms_partinv): S\n");
372 #endif
373 
374  iaux = LR->cols * LR->rows;
375  for(ptr_1 = LR->rel+1, ptr_2 = LR->iel+1, ptr_end = LR->rel+iaux;
376  ptr_1 <= ptr_end; ptr_1 ++, ptr_2 ++)
377  {
378  *ptr_1 = -*ptr_1;
379  *ptr_2 = -*ptr_2;
380  }
381 
382  matfree(Maux_a);
383  matfree(Maux_b);
384 
385 /*************************************************************************
386  insert UL (upper left), UR (upper right), LL (lower left), and
387  LR (lower right) into Minv.
388 *************************************************************************/
389 
390 #ifdef CONTROL_X
391  fprintf(STDCTR,"(ms_partinv): insert\n");
392 #endif
393 
394  Minv = matalloc(Minv, Mbg->rows, Mbg->cols, NUM_COMPLEX);
395 
396  iaux = first_atoms * (l_max + 1)*(l_max + 1) + 1;
397  Minv = matins(Minv, UL, 1, 1);
398  Minv = matins(Minv, UR, 1, iaux);
399  Minv = matins(Minv, LL, iaux, 1);
400  Minv = matins(Minv, LR, iaux, iaux);
401 
402  matfree(UL);
403  matfree(UR);
404  matfree(LL);
405  matfree(LR);
406 
407  return(Minv);
408 } /* end of function ms_partinv */
