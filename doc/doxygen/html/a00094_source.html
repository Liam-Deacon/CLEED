508 coord *lattice_get_surface_normal(const lattice *lat, const coord *a1,
509  const coord *a2, const coord *a3)
510 {
511  double h = lat->vec_h;
512  double k = lat->vec_k;
513  double l = lat->vec_l;
514 
515 
516  #if DEBUG > 0
517  fprintf(stderr, "** debug (lattice_get_surface_normal): malloc");
518  #endif
519 
520  coord *normal = (coord*) malloc(sizeof(coord));
521 
522  if (normal == NULL) return NULL;
523 
524  #if DEBUG > 0
525  fprintf(stderr, " - success \n");
526  #endif
527 
528  normal = coord_init();
529 
530  /* h, k, l not all 0 */
531  if( (fabs(h) > TOLERANCE) ||
532  (fabs(k) > TOLERANCE) ||
533  (fabs(l) > TOLERANCE) )
534  {
535  normal->x = (h*(a2->y*a3->z - a2->z*a3->y) +
536  k*(a3->y*a1->z - a3->z*a1->y) +
537  l*(a1->y*a2->z - a1->z*a2->y) );
538 
539  normal->y = (h*(a2->z*a3->x - a2->x*a3->z) +
540  k*(a3->z*a1->x - a3->x*a1->z) +
541  l*(a1->z*a2->x - a1->x*a2->z) );
542 
543  normal->z = (h*(a2->x*a3->y - a2->y*a3->x) +
544  k*(a3->x*a1->y - a3->y*a1->x) +
545  l*(a1->x*a2->y - a1->y*a2->x) );
546  }
547  else /* h, k, l all 0 */
548  {
549  fprintf(stderr, "*** error (latt): all three Miller indices are zero.\n");
550  exit(1);
551  }
552 
553  #if DEBUG > 0
554  fprintf(stderr, "** debug (lattice_get_surface_normal): return\n");
555  #endif
556 
557  return (normal);
558 }

560 void lattice_setup(lattice *lat, coord *a1, coord *a2,
561  coord *a3, coord *nor, coord *bas, char *bas_name, size_t *n_bas)
562 {
563 
564  *n_bas = 1;
565  double a = lat->a_latt;
566  double c = lat->c_latt;
567  double h = lat->vec_h;
568  double k = lat->vec_k;
569  double l = lat->vec_l;
570 
571  if(lat->latt_type == LAT_FCC)
572  {
573 
574  /********************************************************
575  LAT_FCC
576  ********************************************************/
577 
578  strncpy(bas_name, lat->atoms[0].element, NAMSZ);
579 
580  coord_set(a1, 0.5 * a, 0.0, 0.5 * a);
581  coord_set(a2, 0.5 * a,0.5 * a, 0.0);
582  coord_set(a3, 0.0, 0.5 * a, 0.5 * a);
583 
584  lat->a_nn = a / sqrt(2.);
585 
586  /*** Find surface normal ***/
587  coord_set(nor, h * a, k * a, l * a);
588 
589  } /* if fcc */
590 
591  else if(lat->latt_type == LAT_HCP)
592  {
593  /********************************************************
594  LAT_HCP
595  ********************************************************/
596 
597  *n_bas = 2;
598  realloc((coord*) bas, sizeof(coord) * (*n_bas));
599  realloc((char *) bas_name, *n_bas * NAMSZ * sizeof(char));
600 
601  strncpy(bas_name, lat->atoms[0].element, NAMSZ);
602  strncpy(bas_name + NAMSZ, lat->atoms[0].element, NAMSZ);
603 
604  coord_set(&bas[0], 0., 0., 0.);
605  coord_set(&bas[1], 0., 0., 0.);
606 
607  /* value for Ru */
608  if (fabs(c) < TOLERANCE)
609  {
610  lat->c_latt = 1.59 * a;
611  c = lat->c_latt;
612  }
613 
614  coord_set(a1, 0.5 * a, -sqrt(0.75) * a, 0.);
615  coord_set(a2, 0.5 * a, sqrt(0.75) * a, 0.);
616  coord_set(a3, 0., 0., c);
617 
618  coord_set(bas+sizeof(coord), 0.0, sqrt(1./3.) * a, 0.5 * c);
619 
620  lat->a_nn = a;
621 
622  /*** Find surface normal ***/
623  coord_set(nor, sqrt(0.75) * c * (k + h), 0.5 * c * (k - h), sqrt(0.75) * a * l);
624 
625  } /* if hcp */
626 
627  else if(lat->latt_type == LAT_BCC)
628  {
629  /********************************************************
630  LAT_BCC
631  ********************************************************/
632  strncpy (bas_name, lat->atoms[0].element, NAMSZ);
633 
634  coord_set(a1, a, 0., 0.);
635  coord_set(a2, 0., a, 0.);
636  coord_set(a3, 0.5 * a, 0.5 * a, 0.5 * a);
637 
638  lat->a_nn = a * sqrt(0.75);
639 
640  /*** Find surface normal ***/
641  coord_set(nor, (h * a), (k * a), (l * a));
642 
643  } /* if bcc */
644 
645  else if(lat->latt_type == LAT_DIA)
646  {
647  /********************************************************
648  LAT_DIA
649  ********************************************************/
650 
651  *n_bas = 2;
652  realloc((coord*) bas, sizeof(coord) * (*n_bas));
653  realloc((char *) bas_name, *n_bas * NAMSZ * sizeof(char));
654 
655  strncpy (bas_name, lat->atoms[0].element, NAMSZ);
656  strncpy (bas_name + NAMSZ, lat->atoms[0].element, NAMSZ);
657 
658  /* value for Ru */
659  if (fabs(lat->c_latt) < TOLERANCE)
660  {
661  lat->c_latt = 1.59 * a;
662  c = lat->c_latt;
663  }
664 
665  coord_set(a1, 0.5 * a, 0.0, 0.5 * a);
666  coord_set(a2, 0.5 * a, 0.5 * a, 0.0);
667  coord_set(a3, 0.0, 0.5 * a, 0.5 * a);
668 
669  coord_set(&bas[0], 0., 0., 0.);
670  coord_set(&bas[1], 0.25 * a, 0.25 * a, 0.25 * a);
671 
672  lat->a_nn = a * sqrt(0.375);
673 
674  } /* if dia */
675 
676  else if(lat->latt_type == LAT_INP)
677  {
678  /********************************************************
679  LAT_INP: external lattice input
680  ********************************************************/
681 
682  lattice_read(lat, a1, a2, a3, nor, bas, bas_name, &*n_bas);
683 
684  #if DEBUG > 0
685  fprintf(stderr, "** debug (lattice_read): return\n");
686  #endif
687 
688  } /* if LAT_INP */
689 
690 /********************************************************
691  ELSE
692 ********************************************************/
693  else
694  {
695  fprintf(stderr, "*** error (lattice_setup): "
696  "bad lattice type or input file\n");
697  }
698 
699  /* h, k, l all 0 */
700  if( ( fabs(lat->vec_h - 0.) < (TOLERANCE * fabs(lat->vec_h)) ) &&
701  ( fabs(lat->vec_k - 0.) < (TOLERANCE * fabs(lat->vec_k)) ) &&
702  ( fabs(lat->vec_l - 0.) < (TOLERANCE * fabs(lat->vec_l)) ) )
703  {
704  fprintf(stderr, "***error (lattice_setup): all three Miller indices are zero.\n");
705  exit(1);
706  }
707 
708  #if (DEBUG == 1)
709  fprintf(stderr, "** debug (lattice_setup): finished\n");
710  #endif
711 
712 }

729 void lattice_read(lattice *lat, coord *a1, coord *a2, coord *a3,
730  coord *nor, coord *bas, char *bas_name, size_t *n_bas)
731 {
732 
733  FILE *inp_stream = stdin;
734  coord faux = {.x=0., .y=0., .z=0.};
735  size_t i_bas = 0;
736  coord b1, b2;
737  double R[2][2];
738  char line_buffer[STRSZ];
739  char faux_name[NAMSZ];
740 
741  /* Translate basis vectors such that all z are < 0. */
742  faux.x = bas[0].x;
743  faux.y = bas[0].y;
744  faux.z = bas[0].z;
745  if (bas_name != NULL)
746  {
747  strncpy(faux_name, bas_name, NAMSZ);
748  }
749 
750  fprintf(stderr, "%s -> %s\n", bas_name, faux_name);
751 
752  if (strcmp(lat->input_filename, "stdin") != 0)
753  {
754  if ((inp_stream = fopen(lat->input_filename, "r")) == NULL)
755  {
756  fprintf(stderr, "*** error (lattice_read): open failed %s\n",
757  lat->input_filename);
758  exit(1);
759  }
760  fprintf(inf_stream, "Read lattice input from file \"%s\" \n",
761  lat->input_filename);
762  fprintf(ctr_stream, "Read lattice input from file \"%s\" \n",
763  lat->input_filename);
764  }
765 
766  *n_bas = MAX_INP_ATOMS;
767 
768  realloc((coord*) bas, sizeof(coord) * (*n_bas));
769  realloc((char *) bas_name, *n_bas * NAMSZ * sizeof(char));
770 
771  R[0][0] = R[1][1] = 1.;
772  R[0][1] = R[1][0] = 0.;
773 
774  while((fgets(line_buffer, STRSZ, inp_stream) != NULL) && (i_bas < *n_bas))
775  {
776 
777  fprintf(inf_stream, "%s", line_buffer);
778  #if DEBUG == 1
779  fprintf(ctr_stream, "%s", line_buffer);
780  #endif
781 
782  /* read lattice vectors and superstructure matrix */
783  if( strncmp(line_buffer, "a1:", 3) == 0 )
784  {
785  sscanf(line_buffer+3, "%lf %lf %lf", &a1->x, &a1->y, &a1->z);
786  }
787  if( strncmp(line_buffer, "a2:", 3) == 0 )
788  {
789  sscanf(line_buffer+3, "%lf %lf %lf", &a2->x, &a2->y, &a2->z);
790  }
