247 double **normal_get_rotation_matrix(const coord *normal)
248 {
249  double faux_x;
250  double normal_len;
251  double cos_t, cos_p, sin_t, sin_p;
252  double **R = malloc(sizeof(double *) * 3);
253 
254  if (R == NULL) return NULL;
255 
256  R[0] = (double*) malloc(sizeof(double) * 3);
257  R[1] = (double*) malloc(sizeof(double) * 3);
258  R[2] = (double*) malloc(sizeof(double) * 3);
259 
260  normal_len = COORD_MAGNITUDE(normal);
261  cos_t = normal->z / normal_len;
262  sin_t = sqrt ( 1 - cos_t*cos_t);
263 
264  faux_x = sqrt(normal->x*normal->x + normal->y*normal->y);
265 
266  if (fabs(faux_x) > TOLERANCE)
267  {
268  cos_p = normal->x / faux_x;
269  sin_p = normal->y / faux_x;
270  }
271  else
272  {
273  cos_p = 1.;
274  sin_p = 0.;
275  }
276 
277  R[0][0] = cos_t * cos_p;
278  R[0][1] = cos_t * sin_p;
279  R[0][2] = -sin_t;
280 
281  R[1][0] = -sin_p;
282  R[1][1] = cos_p;
283  R[1][2] = 0.0;
284 
285  R[2][0] = sin_t * cos_p;
286  R[2][1] = sin_t * sin_p;
287  R[2][2] = cos_t;
288 
289  return (R);
290 }

292 basis *basis_rotate_parallel_to_x_axis_rhs(const basis *_basis, double **R)
293 {
294  double faux_x;
295  basis *rot_basis = basis_init();
296 
297  faux_x = sqrt(_basis->a[0].x*_basis->a[0].x + _basis->a[0].y*_basis->a[0].y);
298 
299  R[0][0] = _basis->a[0].x / faux_x;

352 {
353  const coord *a1 = BASIS_VECTOR_PTR(basis, 0);

365  normal->x = (hkl->h*(a2->y*a3->z - a2->z*a3->y) +

367  hkl->l*(a1->y*a2->z - a1->z*a2->y) );
