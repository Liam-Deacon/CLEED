44 real rf_rp(const real *eng, const real *e_int, const real *t_int, real vi)
45 {
46  size_t i_eng;
47  real rf_sum, exp_y_sum, the_y_sum;
48  real Y_exp, Y_the, L, e_step;
49 
50  /* compute integrals */
51  exp_y_sum = 0.;
52  the_y_sum = 0.;
53  rf_sum = 0.;
54  for(i_eng = 1;
55  ! IS_EQUAL_REAL(eng[i_eng], F_END_OF_LIST); i_eng ++)
56  {
57  e_step = eng[i_eng] - eng[i_eng-1];
58 
59  /* theoretical Y function */
60  L = (real)(( t_int[i_eng] - t_int[i_eng-1] ) /
61  ( e_step * 0.5 * (t_int[i_eng] + t_int[i_eng-1]) ));
62 
63  Y_the = (real)(L / ( 1. + L*L*vi*vi));
64  the_y_sum += SQUARE(Y_the)*e_step;
65 
66  /* experimental Y function */
67  L = (real)((e_int[i_eng] - e_int[i_eng-1]) /
68  ( e_step * 0.5 * (e_int[i_eng] + e_int[i_eng-1]) ));
69 
70  Y_exp = (real)(L / ( 1. + L*L*vi*vi));
71  exp_y_sum += SQUARE(Y_exp)*e_step;
72 
73  /* compute R-factor sum */
74  rf_sum += SQUARE(Y_the - Y_exp)*e_step;
75  }
76 
77  #ifdef CONTROL
78  fprintf(STDCTR, "(cr_rp): exp_sum: %f the_sum: %f rf_sum: %f\n",
79  exp_y_sum, the_y_sum, rf_sum);
80  #endif
81 
82  /*
83  * Divide rf_sum by the normalizing factor (exp_y_sum + the_y_sum)
84  * and return result
