40 void sr_er_gsl(size_t n_dim, real dpos, char *log_file)
41 {
42 
43  char *line_buffer;
44 
45  int iaux = 0;
46  size_t i_par, j_par;
47  size_t mpar = n_dim+1;
48 
49  real y_0;
50 
51  gsl_vector *x = gsl_vector_alloc(n_dim);
52  gsl_vector *x_0 = gsl_vector_calloc(n_dim);
53  gsl_vector *y = gsl_vector_alloc(n_dim);
54  gsl_vector *err = gsl_vector_alloc(n_dim);
55  gsl_vector *del = gsl_vector_alloc(n_dim);
56 
57  real faux, pref, rtol, rr, dr, rfac, rdel=0.;
58 
59  FILE *io_stream, *log_stream;
60 
61 /***********************************************************************
62  write title to log file
63 ***********************************************************************/
64 
65  if( (log_stream = fopen(log_file, "a")) == NULL) { OPEN_ERROR(log_file); }
66  fprintf(log_stream, "=> DETERMINE ERROR BARS:\n\n");
67 
68 /***********************************************************************
69  allocate memory and preset variables
70 ***********************************************************************/
71 
72  line_buffer = (char *) malloc(STRSZ * sizeof(char));
73 
74  mpar = mpar * 1; /* set but not used */
75 
76 /***********************************************************************
77  Calculate R factor for minimum.
78 ***********************************************************************/
79 
80  fprintf(log_stream, "=> displace parameters from minimum:\n");
81  fclose(log_stream);
82 
83  y_0 = sr_evalrf_gsl(x_0);
84 
85 /***********************************************************************
86  Read R factor value and RR factor for minimum
87 ***********************************************************************/
88 
89  sprintf(line_buffer, "%s.dum", sr_project);
90  io_stream = fopen(line_buffer, "r");
91 
92  while( fgets(line_buffer, STRSZ, io_stream) != NULL)
93  {
94  if(
95  #ifdef REAL_IS_DOUBLE
96  (iaux = sscanf(line_buffer, "%lf %lf", &rfac, &rr) )
97  #endif
98  #ifdef REAL_IS_FLOAT
99  (iaux = sscanf(line_buffer, "%f %f", &rfac, &rr) )
100  #endif
101  == 2) break;
102  }
103 
104 /* Stop with error message if reading error */
105  if( iaux != 2)
106  {
107  log_stream = fopen(log_file, "a");
108  fprintf(log_stream, "*** error (sr_er_gsl):"
109  " failed to read output from '%s'\n", getenv("CSEARCH_RFAC"));
110  fclose(log_stream);
111  exit(1);
112  }
113 
114  fclose (io_stream);
115 
116  pref = y_0 * rr;
117  rtol = R_TOLERANCE / rr;
118 
119  #ifdef CONTROL
120  fprintf(STDCTR, "(sr_er_gsl):"
121  " rfac = %.4f, rr = %.4f, rdel = %.4f\n", rfac, rr, rdel);
122  #endif
123 
124 /***********************************************************************
125  Calculate R factors for displacements
126 ***********************************************************************/
127 
128  for (i_par = 1; i_par <= n_dim; i_par ++)
129  {
130 
131  gsl_vector_set(del, i_par-1, dpos);
132  rdel = 1.;
133 
134  do
135  {
136 
137  if ( rdel < rtol )
138  {
139  if( rdel < 0. )
140  {
141  #ifdef ERROR
142  fprintf(STDERR, "*** error (sr_er_gsl):"
143  " minimum not found (%d)\n", i_par);
144  #endif
145  exit(1);
146  }
147  else
148  {
149  #ifdef WARNING
150  fprintf(STDWAR, "* warning (sr_er_gsl):"
151  " minimum R factor independent of par. %d\n", i_par);
152  #endif
153  rdel = 1.;
154  }
155  }
156  else
157  {
158  gsl_vector_set(del, i_par-1, gsl_vector_get(del, i_par-1) / R_sqrt(rdel));
159 
160  for (j_par = 1; j_par <= n_dim; j_par ++)
161  {
162  if(i_par == j_par)
163  {
164  gsl_vector_set(x, j_par-1, gsl_vector_get(x_0, j_par-1)
165  + gsl_vector_get(del, j_par-1));
166  }
167  else
168  gsl_vector_set(x, j_par-1, gsl_vector_get(x_0, j_par-1));
169  }
170 
171  #ifdef CONTROL
172  fprintf(STDCTR, "(sr_er_gsl):"
173  " Calculate function for parameter (%d)\n", i_par);
174  #endif
175 
176  gsl_vector_set(y, i_par-1, sr_evalrf_gsl(x));
177 
178  rdel = R_fabs(gsl_vector_get(y, i_par-1) - y_0) / pref;
179  }
180 
181  }
182  while( ( rdel < 0.75 ) || ( rdel > 1.25) );
183 
184  } /* for i_par */
185 
186 
187 /***********************************************************************
188  Calculate error bars and write to log file
189 ***********************************************************************/
190 
191  pref = gsl_vector_get(y, 0) * rr;
192  for (i_par = 1; i_par <= n_dim; i_par ++)
193  {
194  gsl_vector_set(del, i_par-1, R_fabs(gsl_vector_get(del, i_par-1)) );
195  }
196 
197  if( (log_stream = fopen(log_file, "a")) == NULL) { OPEN_ERROR(log_file); }
198 
199  fprintf(log_stream, "\n=> ERROR BARS:\n\n");
200  for (i_par = 1; i_par <= n_dim; i_par ++)
201  {
202  dr = gsl_vector_get(y, i_par-1) - y_0;
203  if (dr < 0.)
204  {
205  #ifdef WARNING
206  fprintf(STDWAR, "* warning (sr_er_gsl):"
207  " not at minimum: dr = %.4f\n", dr);
208  #endif
209  dr = R_fabs(dr);
210  }
211 
212  if( ! IS_EQUAL_REAL(dr, 0.))
213  {
214  faux = R_sqrt(pref/dr) * gsl_vector_get(del, i_par-1);
215  gsl_vector_set(err, i_par-1, faux);
216  fprintf(log_stream, "%2d: del R = %.4f; del par = %.4f\n",
217  i_par, dr, faux);
218  }
219  else
220  {
221  gsl_vector_set(err, i_par-1, -1.);
222  fprintf(log_stream, "%2d: del R = %.4f; del par ** undefined **\n",
223  i_par, dr );
224  }
225 
226  } /* for i_par */
227 
228  fclose(log_stream);
229 
230  /* free memory */
231  gsl_vector_free(y);
232  gsl_vector_free(x);
233  gsl_vector_free(x_0);
234  gsl_vector_free(del);
235  gsl_vector_free(err);
236 
237 } /* end of function sr_er */
