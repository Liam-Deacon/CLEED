36 {
37  FILE* file;
38  long int size;
39  char* content;
40 
41  file = fopen(filename, "r");
42 
43  if(file == NULL) return NULL;
44 
45  fseek(file, 0, SEEK_END);
46 
47  size = ftell(file);
48  rewind(file);
49 
50  content = calloc(size + 1, 1);
51
