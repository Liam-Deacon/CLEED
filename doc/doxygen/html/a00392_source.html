69 /*
70  * TIFF Image File Directories are comprised of
71  * a table of field descriptors of the form shown
72  * below. The table is sorted in ascending order
73  * by tag. The values associated with each entry
74  * are disjoint and may appear anywhere in the file
75  * (so long as they are placed on a word boundary).
76  *
77  * If the value is 4 bytes or less, then it is placed
78  * in the offset field to save space. If the value
79  * is less than 4 bytes, it is left-justified in the
80  * offset field.
81  */
82 typedef struct {
83  uint16 tdir_tag; /* see below */
84  uint16 tdir_type; /* data type; see below */
85  uint32 tdir_count; /* number of items; length in spec */
86  uint32 tdir_offset; /* byte offset to field data */
87 } TIFFDirEntry;

99  * Note: RATIONALs are the ratio of two 32-bit integer values.

101 typedef enum {
102  TIFF_NOTYPE = 0, /* placeholder */
103  TIFF_BYTE = 1, /* 8-bit unsigned integer */
104  TIFF_ASCII = 2, /* 8-bit bytes w/ last byte null */
105  TIFF_SHORT = 3, /* 16-bit unsigned integer */
106  TIFF_LONG = 4, /* 32-bit unsigned integer */
107  TIFF_RATIONAL = 5, /* 64-bit unsigned fraction */
108  TIFF_SBYTE = 6, /* !8-bit signed integer */
109  TIFF_UNDEFINED = 7, /* !8-bit untyped data */
110  TIFF_SSHORT = 8, /* !16-bit signed integer */
111  TIFF_SLONG = 9, /* !32-bit signed integer */
112  TIFF_SRATIONAL = 10, /* !64-bit signed fraction */
113  TIFF_FLOAT = 11, /* !32-bit IEEE floating point */
114  TIFF_DOUBLE = 12 /* !64-bit IEEE floating point */
115 } TIFFDataType;
