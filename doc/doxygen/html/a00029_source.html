23  calcoi.c recalculates the center of each spot using the
24  center of gravity method within the disc: center = position found by fimax
25  radius = range
26  if spot[i].control -bit SPOT_GOOD_S2N is set, the calculation is suppressed.

32  spot.yy/xx = the calculated spot-positions

47 /* loop over all apots in list spot */

49  for( i=0; i<nspot; i++)

59  high = MIN( h0+(int)range , cols-1 );

64 /* loop over integration area rectangle */
65 
66  hsum = 0;
67  vsum = 0;
68  valsum = 1;

70  for( v=lowv; v<=higv; v++)
71  {
72  for( h=lowh; h<=high; h++)

75  if ( pos >= cols*rows || pos < 0 ) continue; /* out of frame */
76  val = im[ pos ];
77  hsum += val * (h-h0);
78  vsum += val * (v-v0);
79  valsum += val;
80  }
81  }

84  mean(x) = sum(val * x) / sum(val) */
