38 int r_ludcmp( real *a, int * indx, int n )
39 
40 /*
41  Perform LU decomposition (first step of matrix inversion).
42 
43  parameters:
44  a - input: pointer to the matrix to be inverted (will be modified).
45  indx - output: int vector which records the row permutation effected
46  by the partial pivoting (must be allocated with the right size).
47  n - input: dimension of matrix
48 
49  return value:
50  +/- 1 depending on whether the number of row interchanges was
51  even or odd, respectively.
52  0 if failed.
53 
54  Given a NxN matrix A this routine replaces it by the LU decomposition
55  of a rowwise permutation of itself. A is output in LU arrangement
56  (NR eq. 2.3.14 );
57  indx is an output vector which records the row permutation effected by
58  the partial pivoting.
59  The return value is +/- 1 depending on whether the number of row
60  interchanges was even or odd, respectively. This routine is used in
61  combination with "lubksb" to solve linear equations.
62 */
63 
64 {
65 int d;
66 
67 int i_r, i_c;
68 int i_max;
69 
70 real big, dum, sum;
71 real *vv; /* vv stores the implicit scaling of each row */
72 real *ptr1, *ptr2, *ptr_end; /* used as counters in the innermost loops */
73 
74  vv = (real *)malloc( (n+1) * sizeof(real) );
75  d = i_max = 1;
76 
77 /* get implicit scaling information */
78  for (i_r = 1; i_r <= n; i_r ++ ) /* loop over rows */
79  {
80  big = 0.0;
81  /* loop over cols */
82  for ( ptr1 = a + (i_r - 1)*n + 1, ptr_end = a + (i_r -1)*n + n;
83  ptr1 <= ptr_end; ptr1 ++)
84  if ((dum = fabs(*ptr1)) > big) big = dum;
85  if (IS_EQUAL_REAL(big, 0.0))
86  {
87 #ifdef ERROR
88  fprintf(STDERR," *** error (ludcmp): Singular matrix \n");
89 #endif
90  return(0);
91  }
92  vv[i_r]=1.0/big;
93  }
94  for (i_c = 1; i_c <= n; i_c++) /* loop over columns in Crout's method */
95  { /* NR eq 2.3.12 except for i_r == i_c */
96  for (i_r = 1; i_r < i_c; i_r ++)
97  {
98  sum = *(a + (i_r - 1)*n + i_c);
99 
100  for (ptr1 = a + (i_r - 1)*n + 1,
101  ptr2 = a + i_c,
102  ptr_end = a + (i_r - 1)*n + i_r ;
103  ptr1 < ptr_end; ptr1 ++, ptr2 += n)
104  sum -= (*ptr1) * (*ptr2);
105 
106  *(a + (i_r - 1)*n + i_c) = sum;
107  } /* for i_r */
108 /*
109  Initialize for the search for largest pivot element
110  This is i_r = i_c for NR eq 2.3.12 and
111  i_r = i_c +1, ..., n for NR eq 2.3.13
112 */
113  big = 0.0;
114  for (i_r = i_c; i_r <= n; i_r ++)
115  {
116  sum = *(a + (i_r - 1)*n + i_c);
117 
118  for (ptr1 = a + (i_r - 1)*n + 1,
119  ptr2 = a + i_c,
120  ptr_end = a + (i_r - 1)*n + i_c;
121  ptr1 < ptr_end; ptr1 ++, ptr2 += n)
122  sum -= (*ptr1) * (*ptr2);
123 
124  *(a + (i_r - 1)*n + i_c) = sum;
125 
126 /*
127  Figure of the merit for the pivot - is it better than the best so far?
128 */
129  if ( ( dum = vv[i_r]*fabs(sum) ) >= big)
130  {
131  big = dum;
132  i_max = i_r;
133  }
134  } /* for i_r */
135 
136  if (i_c != i_max) /* Do we need to interchange rows? */
137  {
138  for (ptr1 = a + (i_max - 1)*n + 1,
139  ptr2 = a + (i_c - 1)*n + 1,
140  ptr_end = a + i_max*n;
141  ptr1 <= ptr_end; ptr1 ++, ptr2 ++)
142  {
143  dum = *ptr1;
144  *ptr1 = *ptr2;
145  *ptr2 = dum;
146  }
147  d = -d; /* change the parity of d */
148  vv[i_max]=vv[i_c]; /* Also interchange scale factor */
149  } /* if i_c != i_max */
150  indx[i_c]=i_max;
151 /*
152  If the pivot element is zero the matrix is singular (at least to the
153  precision of the algorithm). For some applications on singular matrices
154  it is desirable to substitute TINY to zero.
155 */
156  if ( *(a + (i_c - 1)*n + i_c) == 0.0 ) *(a + (i_c - 1)*n + i_c) = TINY;
157 
158 /*
159  Now, finally, divide by the pivot element.
160 */
161  if (i_c != n)
162  {
163  dum=1.0/(*(a + (i_c - 1)*n + i_c));
164 
165  for (ptr1 = a + i_c*n + i_c, ptr_end = a + n*n + i_c;
166  ptr1 < ptr_end; ptr1 += n)
167  *ptr1 *= dum;
168  } /* if i_c != n */
169  } /* for i_c */
170 
171  free(vv);
172  return(d);
173 } /* end of function r_ludcmp */

181 int r_luinv(real * inv, real * lu, int * indx, int n)
182 
183 /*
184  Solves the set of n linear equations A*A_1 = I.
185 
186  parameters:
187 
188  inv - (output) pointer to the inverse matrix. Enough memory must be
189  allocated for this pointer (n*n +1).
190  lu - input: pointer to the matrix to be inverted (as LU decomposition
191  as returned by r_ludcmp).
192  indx - input: int vector which records the row permutation (as returned
193  by ludcmp)
194  n - input: dimension of matrix
195 
196  return value:
197  1 if o.k.
198  0 if failed.
199 
200  Here A is input but not as the matrix A but rather as its LU decomposition
201  determined by the routine ludcmp. indx is input as the permutation vector
202  returned by ludcmp. b is input as the right hand side vector b, and returns
203  with the solution vector X. A and index are not modified by this routine
204  and can be left in place for successive calls with different right-hand
205  sides b. This routine takes into account the possibility that b will begin
206  with many zero elements, so it is efficient for the use in matrix inversion.
207 */
208 
209 {
210 int i_c, i_i, i_r;
211 
212 real sum;
213 real *ptr1, *ptr2, *ptr_end; /* pointers used in innermost loops */
214 
215 /*
216  First, preset inv as nxn identity matrix.
217 */
218 
219  for (ptr1 = inv + 1, ptr_end = inv + n*n; ptr1 <= ptr_end; ptr1 ++)
220  *ptr1 = 0.;
221  for (ptr1 = inv + 1, ptr_end = inv + n*n; ptr1 <= ptr_end; ptr1 += n+1)
222  *ptr1 = 1.0;
223 
224 /*
225  Backsubstitution starts here.
226 */
227  for (i_c = 1; i_c <= n; i_c ++) /* loop over columns of inv */
228  {
229  i_i = 0;
230  for (i_r = 1; i_r <= n; i_r ++)
231  /* loop through columns of inv (i.e. over rows) */
232  {
233  sum = *(inv + (indx[i_r] - 1)*n + i_c);
234  *(inv + (indx[i_r] - 1)*n + i_c) = *(inv + (i_r - 1)*n + i_c);
235 
236  if (i_i)
237  {
238  for (ptr1 = lu + (i_r - 1)*n + i_i,
239  ptr2 = inv + (i_i - 1)*n + i_c,
240  ptr_end = lu + (i_r - 1)*n + i_r;
241  ptr1 < ptr_end; ptr1 ++, ptr2 += n )
242  sum -= (*ptr1) * (*ptr2);
243  }
244  else if ( ! IS_EQUAL_REAL(sum, 0.0)) i_i = i_r;
245 
246  *(inv + (i_r - 1)*n + i_c) = sum;
247  }
248 
249  for (i_r = n; i_r >= 1; i_r --)
250  {
251  sum = *(inv + (i_r - 1)*n + i_c);
252  for (ptr1 = lu + (i_r - 1)*n + i_r + 1,
253  ptr2 = inv + i_r*n + i_c,
254  ptr_end = lu + (i_r - 1)*n + n;
255  ptr1 <= ptr_end; ptr1 ++, ptr2 += n )
256  sum -= (*ptr1) * (*ptr2);
257 
258  *(inv + (i_r - 1)*n + i_c) = sum/(*(lu + (i_r - 1)*n + i_r));
259  }
260  } /* for i_c */
261  return(1);
262 } /* end of function r_luinv */

266 real * r_lubksb(real * lu, int * indx, real * br, int n)
267 
268 /*
269  Solves the set of n linear equations A*X = b.
270 
271  parameters:
272 
273  lu - input: pointer to the matrix to be inverted (as LU decomposition
274  as returned by r_ludcmp).
275  br - input: pointer to the vector to be multiplied with the inverted
276  matrix (will be overwritten).
277  indx - input: int vector which records the row permutation (as returned
278  by ludcmp)
279  n - input: dimension of matrix
280 
281  return value:
282  pointer to the result.
283  NULL, if failed.
284 
285  Here A is input but not as the matrix A but rather as its LU decomposition
286  determined by the routine ludcmp. indx is input as the permutation vector
287  returned by ludcmp. b is input as the right hand side vector b, and returns
288  with the solution vector X. A and index are not modified by this routine
289  and can be left in place for successive calls with different right-hand
290  sides b. This routine takes into account the possibility that b will begin
291  with many zero elements, so it is efficient for the use in matrix inversion.
292 */
293 
294 {
295 int i_c, i_i, i_r;
296 
297 real sum;
298 real *ptr1, *ptr2, *ptr_end; /* pointers used in innermost loops */
299 
300  for (i_c = 1; i_c <= n; i_c ++) /* loop over columns of br */
301  {
302  i_i = 0;
303  for (i_r = 1; i_r <= n; i_r ++)
304  /* loop through columns of br (i.e. over rows) */
305  {
306  sum = *(br + (indx[i_r] - 1)*n + i_c);
307  *(br + (indx[i_r] - 1)*n + i_c) = *(br + (i_r - 1)*n + i_c);
308 
309  if (i_i)
310  {
311  for (ptr1 = lu + (i_r - 1)*n + i_i,
312  ptr2 = br + (i_i - 1)*n + i_c,
313  ptr_end = lu + (i_r - 1)*n + i_r;
314  ptr1 < ptr_end; ptr1 ++, ptr2 += n )
315  sum -= (*ptr1) * (*ptr2);
316  }
317  else if ( ! IS_EQUAL_REAL(sum, 0.0)) i_i = i_r;
318 
319  *(br + (i_r - 1)*n + i_c) = sum;
320  }
321 
322  for (i_r = n; i_r >= 1; i_r --)
323  {
324  sum = *(br + (i_r - 1)*n + i_c);
325  for (ptr1 = lu + (i_r - 1)*n + i_r + 1,
326  ptr2 = br + i_r*n + i_c,
327  ptr_end = lu + (i_r - 1)*n + n;
328  ptr1 <= ptr_end; ptr1 ++, ptr2 += n )
329  sum -= (*ptr1) * (*ptr2);
330 
331  *(br + (i_r - 1)*n + i_c) = sum/(*(lu + (i_r - 1)*n + i_r));
332  }
333  } /* for i_c */
334  return(br);
335 } /* end of function r_lubksb */
