34 /************************************************************************
35 
36  Set up a list of all beams used within the energy loop.
37 
38  INPUT:
39 
40  leed_beam_t ** p_beams - (output) pointer to the list of beams to be
41  included at the current energy.
42  leed_cryst_t *c_par - all necessary structural parameters (for details
43  see "leed_def.h").
44  leed_var *v_par - all necessary parameters that change during the
45  energy loop (for details see "leed_def.h").
46  used:
47  real vr - real part of optical potential
48  real theta, phi - incident k-vector.
49  real epsilon - parameter determining the cutoff radius for
50  k_par. (maximum amplitude which can propagate between two
51  layers).
52  real eng_max - maximum energy in the energy loop (in Hartree).
53 
54  DESIGN:
55 
56  The order of the output list is:
57  - increasing modulus of momentum transfer k_par (lowest k_par first)
58  - increasing 1st index (lowest 1st index first for the same k_par)
59  - increasing 2nd index (lowest 2nd index first for the same 1st index
60  and k_par)
61 
62  RETURN VALUE:
63 
64  int n_set - number of beam sets in the list pointed to by p_beams.
65  -1 if failed
66

91 /**********************************************************************
92  Allocate storage space
93  - Set eng_max to vacuum energy - optical potential
94  - Determine k_max (square of max k_par) from epsilon and dmin.
95  - Determine the max. number of beams within this radius (iaux)
96  and allocate memory for the beam list. (The formula given in
97  VHT p. 24 is not an upper limit. A prefactor 1/(PI*PI)
98  (0.10132118) is saver than 1/(4*PI) ).
99 **********************************************************************/
100 
101  eng_max -= v_par->vr;
102  faux_r = R_log(v_par->epsilon) / c_par->dmin;
103  k_max_2 = faux_r*faux_r + 2*eng_max;
104  k_max = R_sqrt(k_max_2);
105 
106  iaux = 2 + (int)( 0.10132118 * c_par->rel_area_sup * c_par->area * k_max_2 );
107 
108  if (*p_beams == NULL)
109  *p_beams = (leed_beam_t *)calloc(iaux, sizeof(leed_beam_t));
110  else
111  *p_beams = (leed_beam_t *)realloc(*p_beams, iaux*sizeof(leed_beam_t));
112 
113  if(*p_beams == NULL)
114  {
115 #ifdef ERROR

117 #endif
118  exit(1);
119  }

123  }

136  Some often used values:

139  - k_in at max. energy (k_in)

156 #endif

158 /**********************************************************************
159  Determine number of beam sets (n_set)
160  and offsets (store in bm_off)
161  Each beam set is represented exactly once within the first BZ
162  (i.e. within the diamond: (0,0)(1,0)(0,1)(1,1))
163  => raster through the first BZ and store all fractional
164  order beams in the array bm_off.

166

179 #endif

183  {

193  (bm_off+i_set)->k_r[2] = k_x*g1_y + k_y*g2_y;

197  } /* for n1,n2 */

208 /*********************************************************
209  Find the beams within the radius defined by k_max
210  - determine boundaries for beam indices n1 and n2
211  - loop over beam indices.
212 *********************************************************/
213 
214 /* a1 = length of g1 */
215  a1 = R_hypot(g1_x, g1_y);
216 /* a2 = length of g2 */
217  a2 = R_hypot(g2_x, g2_y);
218 
219 /* a2 * cos(a1,a2) */
220  faux_r = R_fabs((g1_x*g2_x + g1_y*g2_y)/a1);
221 /* a2 * sin(a1,a2) */
222  faux_i = R_fabs((g1_x*g2_y - g1_y*g2_x)/a1);
223 
224 /*
225  n2_max = k_max / (sin(a1,a2) * a2) + k_in/a2
226  n1_max = k_max / a1 + n2_max * (cos(a1,a2) *a2)/a1 + k_in/a1
227 */
228  n2_max = 2 + (int)(k_max/faux_i + k_in[0]/a2);
229  n1_max = 2 + (int)( k_max/a1 + n2_max * faux_r/ a1 + k_in[0]/a1);
230 
231 #ifdef CONTROL_X
232  fprintf(STDCTR,"(leed_beam_gen): n1_max = %2d, n2_max = %2d\n", n1_max, n2_max);
233 #endif
234 
235 /*
236  k_r, k_i is now defined by the complex energy
237 */
238  cri_sqrt(&k_r, &k_i, 2.*eng_max, 2.*v_par->eng_i);

240 /**********************************************************************
241  Loop over beam sets
242 **********************************************************************/

244  for(i_set = 0, i_beams = 0, offset = 0;

247  {

253  for(n2 = -n2_max; n2 <= n2_max; n2 ++)
254  {

257

259  if(faux_r <= k_max_2)

262  indices, k_par, k_x/y
263 */

265  (beams + i_beams)->ind_2 = (real)n2 + (bm_off+i_set)->ind_2;
266 
267  k_x = n1*g1_x + n2*g2_x + (bm_off+i_set)->k_r[1];
268  k_y = n1*g1_y + n2*g2_y + (bm_off+i_set)->k_r[2];
269 
270  (beams + i_beams)->k_r[1] = k_x;
271  (beams + i_beams)->k_r[2] = k_y;
272  (beams + i_beams)->k_par = SQUARE(k_x) + SQUARE(k_y);
273 
274  (beams + i_beams)->k_i[1] = 0.;
275  (beams + i_beams)->k_i[2] = 0.;
276 
277  (beams + i_beams)->set = i_set;

282 
283  i_beams ++;
284  } /* if inside k_max */
285  } /* for n1/n2 */
286  /*********************************************************
287  1st pass: Sort the beams according to the parallel component
288  (i.e. smallest k_par first)
289  *********************************************************/
290 #ifdef CONTROL
291  fprintf(STDCTR,"(leed_beam_gen): SORTING %2d beams in set %d:\n",
292  i_beams - offset, i_set);
293 #endif
294  for(n1 = offset; n1 < i_beams; n1 ++)
295  {
296  for(n2 = n1+1; n2 < i_beams; n2 ++)
297  {
298  if((beams + n2)->k_par < (beams + n1)->k_par )
299  {
300  memcpy( & beam_aux, beams + n2, sizeof(leed_beam_t) );
301  memcpy( beams + n2, beams + n1, sizeof(leed_beam_t) );
302  memcpy( beams + n1, & beam_aux, sizeof(leed_beam_t) );
303  }
304  } /* n2 */
305  } /* n1 */
306 
307  /*********************************************************
308  2nd pass: Sort the beams according to the 1st and 2nd index
309  (i.e. smallest indices first)
310  *********************************************************/
311  for(n1 = offset; n1 < i_beams; n1 ++)
312  {
313  for(n2 = n1+1;
314  fabs( (beams + n2)->k_par - (beams + n1)->k_par ) < K_TOLERANCE;
315  n2++)
316  {
317  if((beams + n2)->ind_1 < (beams + n1)->ind_1 )
318  {
319  memcpy( & beam_aux, beams + n2, sizeof(leed_beam_t) );
320  memcpy( beams + n2, beams + n1, sizeof(leed_beam_t) );
321  memcpy( beams + n1, & beam_aux, sizeof(leed_beam_t) );
322  }
323  if( (IS_EQUAL_REAL((beams + n2)->ind_1, (beams + n1)->ind_1 )) &&
324  ((beams + n2)->ind_2 < (beams + n1)->ind_2 ) )
325  {
326  memcpy( & beam_aux, beams + n2, sizeof(leed_beam_t) );
327  memcpy( beams + n2, beams + n1, sizeof(leed_beam_t) );
328  memcpy( beams + n1, & beam_aux, sizeof(leed_beam_t) );
329  }
330  } /* n2 */
331 #ifdef CONTROL
332  fprintf(STDCTR,"%2d: (%6.2f, %6.2f):\t",
333  n1, (beams + n1)->ind_1, (beams + n1)->ind_2);
334  fprintf(STDCTR,"\td_par: %.2f\tk_r: (%5.2f, %5.2f, %5.2f)\n",
335  R_sqrt((beams + n1)->k_par), (beams + n1)->k_r[1],
336  (beams + n1)->k_r[2], (beams + n1)->k_r[3]);
337 #endif
338  } /* n1 */
339  } /* for i_set */
340 
341 /*
342  Set k_par of the last element of the list to the terminating value.
343 */
344  (beams + i_beams)->k_par = F_END_OF_LIST;
345 
346  return(n_set);
