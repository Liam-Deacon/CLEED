25  leed_beam_t *beams, real eng_vac, real *vec_ab )
26 
27 /************************************************************************
28 
29  Calculate only the first column of the reflection matrix R+- (i.e. the
30  amplitudes of the backscattered beams) for a stack of one (super) layer
31  "a" (can be the complete bulk) and a square potential step
32  (z(a) < z(pot. step)) by layer doubling:
33 
34  INPUT:
35 
36  mat Rpm1 - (output) fist column of the reflection matrix (+-)
37 
38  mat Rpm_a - (input) reflection matrix (+-) of the "lower" layer "a".
39 
40  beam_str *beams - (input) information about beams.
41  used: k_r, k_i, k_par.
42  real eng_vac - vacuum energy.
43  real *vec_ab - (input) vector pointing from the origin of layer a to
44  the origin of layer b. The usual convention for vectors is
45  used (x = 1, y = 2, z = 3).
46 
47  DESIGN:
48 
49  z(a) < z(b) => vec_ab[3] > 0 (otherwise no convergence!)
50 
51  Rab+- = Rb+- + (Tb++ P+ Ra+- P-) * (I - Rb-+ P+ Ra+- P-)^(-1) * Tb--
52 
53  whereby Tb and Rb are diagonal with:
54 
55  R = (kc - kv) / (kc + kv)
56  T+ = 2*kc / (kc + kv)
57  T- = 2*kv / (kc + kv)
58 
59  kc = modulus of kz inside the crystal = sqrt(2*Ei - k_par^2)
60  kv = modulus of kz outside the crystal sqrt(2*Ev - k_par^2)
61 
62  RETURN VALUES:
63 
64  mat Rpm1 - fist column of the reflection matrix (+-) (not necessarily
65  equal to the first argument).
66 
67 *************************************************************************/
68 {
69 int k;
70 int n_beams, nn_beams; /* total number of beams */
71 
72 real faux_r, faux_i;
73 real *ptr_r, *ptr_i, *ptr_end;
74 
75 mat kv, Pp, Pm, Maux_a, Maux_b; /* temp. storage space */
76 mat Res; /* result will be copied to Rpm_ab */
77 
78 
79  Res = kv = Pp = Pm = Maux_a = Maux_b = NULL;
80 
81 /*************************************************************************
82  Check arguments:
83 *************************************************************************/
84 
85 /*************************************************************************
86  Allocate memory and set up propagators Pp and Pm.
87 
88  Pp = exp[ i *( k_x*v_ab_x + k_y*v_ab_y + k_z*v_ab_z) ]
89  Pm = exp[-i *( k_x*v_ab_x + k_y*v_ab_y - k_z*v_ab_z) ]
90  = exp[ i *(-k_x*v_ab_x - k_y*v_ab_y + k_z*v_ab_z) ]
91 
92 *************************************************************************/
93  n_beams = Rpm_a->cols;
94  nn_beams = n_beams * n_beams;
95 
96  Pp = matalloc(NULL, n_beams, 1, NUM_COMPLEX );
97  Pm = matalloc(NULL, n_beams, 1, NUM_COMPLEX );
98 
99  for( k = 0; k < n_beams; k++)
100  {
101 
102  faux_r = (beams+k)->k_r[1] * vec_ab[1] +
103  (beams+k)->k_r[2] * vec_ab[2] +
104  (beams+k)->k_r[3] * vec_ab[3];
105  faux_i = (beams+k)->k_i[3] * vec_ab[3];
106 
107  cri_expi(Pp->rel+k+1, Pp->iel+k+1, faux_r, faux_i);
108 
109  faux_r -= 2 * (beams+k)->k_r[3] * vec_ab[3];
110 
111  cri_expi(Pm->rel+k+1, Pm->iel+k+1, -faux_r, faux_i);
112 
113  }
114 /*************************************************************************
115  Calculate kz in vacuum and store the results in kv.
116 *************************************************************************/
117  kv = matalloc(kv, n_beams, 1, NUM_COMPLEX );
118 
119  for( k = 0; k < n_beams; k++)
120  {
121  faux_r = 2*eng_vac - SQUARE((beams+k)->k_par);
122  if (faux_r >= 0.)
123  {
124  kv->rel[k+1] = sqrt(faux_r);
125  kv->iel[k+1] = 0.;
126  }
127  else
128  {
129  kv->rel[k+1] = 0.;
130  kv->iel[k+1] = sqrt(-faux_r);
131  }
132  }
133 
134 /*************************************************************************
135  Prepare (Ra+- P-) (-> Maux_a):
136  Multiply the k-th column of Ra+- with the k-th element of P-.
137 *************************************************************************/
138 
139  Maux_a = matcop(Maux_a, Rpm_a);
140 
141  for(k = 1; k <= n_beams; k ++)
142  {
143  faux_r = *(Pm->rel+k);
144  faux_i = *(Pm->iel+k);
145 
146  ptr_end = Maux_a->rel+nn_beams;
147  for (ptr_r = Maux_a->rel+k, ptr_i = Maux_a->iel+k;
148  ptr_r <= ptr_end; ptr_r += n_beams, ptr_i += n_beams)
149  cri_mul(ptr_r, ptr_i, *ptr_r, *ptr_i, faux_r, faux_i);
150  }
151 
152 /*************************************************************************
153  Calculate I - (Rb-+ P+ Ra+- P-) (-> Maux_b):
154  Multiply the k-th row of Maux_a with the k-th element of P+ and Rpot
155  R = (kc - kv) / (kc + kv)
156  and add unity (I).
157 *************************************************************************/
158 
159  Maux_b = matcop(Maux_b, Maux_a);
160 
161  for(k = 1; k <= n_beams; k ++)
162  {
163  cri_div(&faux_r, &faux_i,
164  (beams+k-1)->k_r[3] - kv->rel[k], (beams+k-1)->k_i[3] - kv->iel[k],
165  (beams+k-1)->k_r[3] + kv->rel[k], (beams+k-1)->k_i[3] + kv->iel[k]);
166  cri_mul(&faux_r, &faux_i, Pp->rel[k], Pp->iel[k], faux_r, faux_i);
167 
168  ptr_end = Maux_b->rel + k*Maux_b->cols;
169  for (ptr_r = Maux_b->rel + (k-1)*Maux_b->cols + 1,
170  ptr_i = Maux_b->iel + (k-1)*Maux_b->cols + 1;
171  ptr_r <= ptr_end; ptr_r ++, ptr_i ++)
172  cri_mul(ptr_r, ptr_i, *ptr_r, *ptr_i, faux_r, faux_i);
173  }
174 
175 /* add unity */
176  for(k = 1; k <= nn_beams; k+= Maux_b->cols + 1)
177  {
178  Maux_b->rel[k] += 1.;
179  }
180 
181 /*************************************************************************
182  (i) Matrix inversion:
183  Maux_b = ( I - (Rb-+ P+ Ra+- P-))^(-1)
184 
185  (ii) Multiply the first row with T--(00) and store the result in Res:
186  Res = ( I - (Rb-+ P+ Ra+- P-))^(-1)k1 * Tb--(00)
187  (First allocate Res)
188 
189 (iii) Prepare Res:
190  Res ->
191  Ra+- P- * ( I - (Rb-+ P+ Ra+- P-))^(-1) * Tb-- = Maux_a * Res
192 *************************************************************************/
193 
194 /* (i) */
195  Maux_b = matinv(Maux_b, Maux_b);
196 
197 /* (ii) */
198 
199 /* T-(00) = 2*kv / (kv + kc) */
200  cri_div(&faux_r, &faux_i,
201  2*kv->rel[1], 2*kv->iel[1],
202  beams->k_r[3] + kv->rel[1], beams->k_i[3] + kv->iel[1]);
203 
204 /* Multiply first column with T-(00) */
205  Res = matcol(Res, Maux_b, 1);
206  for(k = 1; k <= n_beams; k ++)
207  {
208  cri_mul(Res->rel+k, Res->iel+k, Res->rel[k], Res->iel[k], faux_r, faux_i);
209  }
210 
211 /* (iii) */
212  Res = matmul(Res, Maux_a, Res);
213 
214 /*************************************************************************
215  Complete the computation of the matrix product in R+-:
216 
217  (i) Multiply with T++ P+, i.e. multiply the elements of Res with
218  T++ * P+
219  (ii) Add the reflextion of (00) at the potential step:
220  R+- = (- kc + kv) / (kc + kv)
221 *************************************************************************/
222 
223 /* (i) */
224  for(k = 1; k <= n_beams; k ++)
225  {
226  cri_div(&faux_r, &faux_i,
227  2*((beams+k-1)->k_r[3]), 2*((beams+k-1)->k_i[3]),
228  (beams+k-1)->k_r[3] + kv->rel[k], (beams+k-1)->k_i[3] + kv->iel[k]);
229  cri_mul(&faux_r, &faux_i, Pp->rel[k], Pp->iel[k], faux_r, faux_i);
230 
231  cri_mul(Res->rel+k, Res->iel+k, Res->rel[k], Res->iel[k], faux_r, faux_i);
232  }
233 
234 /* (ii) */
235 
236  cri_div(&faux_r, &faux_i,
237  - beams->k_r[3] + kv->rel[1], - beams->k_i[3] + kv->iel[1],
238  beams->k_r[3] + kv->rel[1], beams->k_i[3] + kv->iel[1]);
239 
240  Res->rel[1] += faux_r;
241  Res->iel[1] += faux_i;
242 
243 /*************************************************************************
244  - Write the result to the output pointer
245  - Free temporary storage space,
246  - Return.
247 *************************************************************************/
248 
249  Rpm1_ab = matcop(Rpm1_ab, Res);
250 
251  matfree(Pp);
252  matfree(Pm);
253  matfree(kv);
254  matfree(Maux_a);
255  matfree(Maux_b);
256  matfree(Res);
257 
258  return(Rpm1_ab);
259 }
