31 /************************************************************************
32 
33  Calculate the layer scattering matrices for a stack of two layers
34  "a" and "b" (z(a) < z(b)) by layer doubling:
35 
36  INPUT:
37 
38  mat *p_Tpp_ab - (output) pointer to transmission matrix (++) of the stack "ab".
39  mat *p_Tmm_ab - (output) pointer to transmission matrix (--) of the stack "ab".
40  mat *p_Rpm_ab - (output) pointer to reflection matrix (+-) of the stack "ab".
41  mat *p_Rmp_ab - (output) pointer to reflection matrix (-+) of the stack "ab".
42 
43  mat Tpp_a/b - (input) transmission matrix (++) of the single layer "a"/"b".
44  mat Tmm_a/b - (input) transmission matrix (--) of the single layer "a"/"b".
45  mat Rpm_a/b - (input) reflection matrix (+-) of the single layer "a"/"b".
46  mat Rmp_a/b - (input) reflection matrix (-+) of the single layer "a"/"b".
47 
48  beam_str *beams - (input) information about beams.
49  used: k_r, k_i.
50  real *vec_ab - (input) vector pointing from the origin of layer a to
51  the origin of layer b. The usual convention for vectors is
52  used (x = 1, y = 2, z = 3).
53 
54  DESIGN:
55 
56  z(a) < z(b) => vec_ab[3] > 0 (otherwise no convergence!)
57 
58  Tab++ = (Tb++ P+) * (I - Ra+- P- Rb-+ P+)^(-1) * Ta++
59  Rab-+ = Ra-+ + (Ta-- P- Rb-+ P+) * (I - Ra+- P- Rb-+ P+)^(-1) * Ta++
60 
61  Tab-- = (Ta-- P-) * (I - Rb-+ P+ Ra+- P-)^(-1) * Tb--
62  Rab+- = Rb+- + (Tb++ P+ Ra+- P-) * (I - Rb-+ P+ Ra+- P-)^(-1) * Tb--
63 
64  FUNCTIONS:
65 
66  matcop
67  matmul
68  matinv
69 
70  RETURN VALUES:
71 
72 
73 *************************************************************************/
74 {
75 int k;
76 int n_beams, nn_beams; /* total number of beams */

79 real *ptr_r, *ptr_i, *ptr_end;

94 /*************************************************************************
95  Allocate memory and set up propagators Pp and Pm.
96 
97  Pp = exp[ i *( k_x*v_ab_x + k_y*v_ab_y + k_z*v_ab_z) ]
98  Pm = exp[-i *( k_x*v_ab_x + k_y*v_ab_y - k_z*v_ab_z) ]
99  = exp[ i *(-k_x*v_ab_x - k_y*v_ab_y + k_z*v_ab_z) ]
100 *************************************************************************/
101  n_beams = Tpp_a->cols;
102  nn_beams = n_beams * n_beams;
103 
104  Pp = matalloc(NULL, n_beams, 1, NUM_COMPLEX );
105  Pm = matalloc(NULL, n_beams, 1, NUM_COMPLEX );
106 
107 #ifdef CONTROL
108  fprintf(STDCTR, "(leed_ld_2lay): vec_ab(%.2f %.2f %.2f) = vec_from_last\n",
109  vec_ab[1] * BOHR, vec_ab[2] * BOHR , vec_ab[3] * BOHR);
110 #endif
111 
112  for( k = 0; k < n_beams; k++)
113  {
114 #ifdef CONTROL
115  fprintf(STDCTR, "(leed_ld_2lay): %2d: k_r = %6.3f %6.3f %6.3f, k_i = %6.3f;",
116  k, (beams+k)->k_r[1], (beams+k)->k_r[2], (beams+k)->k_r[3],
117  (beams+k)->k_i[3]);
118 #endif
119 
120  faux_r = (beams+k)->k_r[1] * vec_ab[1] +
121  (beams+k)->k_r[2] * vec_ab[2] +
122  (beams+k)->k_r[3] * vec_ab[3];
123  faux_i = (beams+k)->k_i[3] * vec_ab[3];
124 
125  cri_expi(Pp->rel+k+1, Pp->iel+k+1, faux_r, faux_i);
126 
127  faux_r -= 2. * (beams+k)->k_r[3] * vec_ab[3];
128 
129  cri_expi(Pm->rel+k+1, Pm->iel+k+1, -faux_r, faux_i);
130 
131 #ifdef CONTROL
132  fprintf(STDCTR," Pp = (%6.3f,%6.3f), Pm = (%6.3f,%6.3f)",
133  Pp->rel[k+1], Pp->iel[k+1], Pm->rel[k+1], Pm->iel[k+1]);
134 
135  cri_mul(&faux_r,&faux_i,
136  Pp->rel[k+1],Pp->iel[k+1],Pm->rel[k+1],Pm->iel[k+1]);
137  fprintf(STDCTR,": (%6.3f,%6.3f)\n", faux_r, faux_i);
138 #endif
139  }

141 /*************************************************************************
142  Prepare the quantities (Ra+- P-) and -(Rb-+ P+):
143  Multiply the k-th column of Ra+- / Rb-+ with the k-th element of P-/+.
144 *************************************************************************/
145 
146  Maux_a = matcop(Maux_a, Rpm_a);
147  Maux_b = matcop(Maux_b, Rmp_b);
148 
149  for(k = 1; k <= n_beams; k ++)
150  {
151 
152  faux_r = *(Pm->rel+k);
153  faux_i = *(Pm->iel+k);
154 
155  ptr_end = Maux_a->rel+nn_beams;
156  for (ptr_r = Maux_a->rel+k, ptr_i = Maux_a->iel+k;
157  ptr_r <= ptr_end; ptr_r += n_beams, ptr_i += n_beams)
158  cri_mul(ptr_r, ptr_i, *ptr_r, *ptr_i, faux_r, faux_i);
159 
160  faux_r = - *(Pp->rel+k);
161  faux_i = - *(Pp->iel+k);
162 
163  ptr_end = Maux_b->rel+nn_beams;
164  for (ptr_r = Maux_b->rel+k, ptr_i = Maux_b->iel+k;
165  ptr_r <= ptr_end; ptr_r += n_beams, ptr_i += n_beams)
166  cri_mul(ptr_r, ptr_i, *ptr_r, *ptr_i, faux_r, faux_i);
167 
168  }

170 /*************************************************************************
171  (i) Calculate the quantities
172  -(Ra+- P- Rb-+ P+) = Maux_a * Maux_b (-> Tpp_ab) and
173  -(Rb-+ P+ Ra+- P-) = Maux_b * Maux_a (-> Tmm_ab)
174  and add unity.
175 
176  (ii) Matrix inversion:
177  Tpp_ab = ( I - (Ra+- P- Rb-+ P+))^(-1)
178  Tmm_ab = ( I - (Rb-+ P+ Ra+- P-))^(-1)
179 
180 (iii) Multiply with T++_a or T--_b, respectively and store the matrices in
181  Tpp_ab and Tmm_ab respectively:
182  Tpp_ab = ( I - (Ra+- P- Rb-+ P+))^(-1) * Ta++
183  Tmm_ab = ( I - (Rb-+ P+ Ra+- P-))^(-1) * Tb--
184 
185  (iv) Prepare Rpm_ab and Rmp_ab:
186  Rpm_ab ->
187  Ra+- P- * ( I - (Rb-+ P+ Ra+- P-))^(-1) * Tb-- = Maux_a * Tmm_ab
188  Rmp_ab ->
189  - Rb-+ P+ * ( I - (Ra+- P- Rb-+ P+))^(-1) * Ta++ = Maux_b * Tpp_ab
190  (the sign will be corrected later on)
191 *************************************************************************/
192 
193 /* (i) */
194  Tpp_ab = matmul(Tpp_ab, Maux_a, Maux_b);
195  Tmm_ab = matmul(Tmm_ab, Maux_b, Maux_a);
196 
197  for(k = 1; k <= nn_beams; k+= Maux_b->cols + 1)
198  {
199  Tpp_ab->rel[k] += 1.;
200  Tmm_ab->rel[k] += 1.;
201  }
202 
203 #ifdef CONTROL
204  fprintf(STDCTR,
205 "(leed_ld_2lay): Tpp_ab = matmul(Tpp_ab, Maux_a, Maux_b)\n");
206  matshow(Tpp_ab);
207  fprintf(STDCTR,
208 "(leed_ld_2lay): Tmm_ab = matmul(Tmm_ab, Maux_b, Maux_a)\n");
209  matshow(Tmm_ab);
210 #endif
211 
212 /* (ii) */
213  Tpp_ab = matinv(Tpp_ab, Tpp_ab);
214  Tmm_ab = matinv(Tmm_ab, Tmm_ab);
215 
216 /* (iii) */
217  Tpp_ab = matmul(Tpp_ab, Tpp_ab, Tpp_a);
218  Tmm_ab = matmul(Tmm_ab, Tmm_ab, Tmm_b);
219 
220 
221 /* (iv) */
222  Rpm_ab = matmul(Rpm_ab, Maux_a, Tmm_ab);
223  Rmp_ab = matmul(Rmp_ab, Maux_b, Tpp_ab);
224 
225 /*************************************************************************
226  Prepare the quantities
227  Maux_a = (Ta-- P-) and Maux_b = (Tb++ P+):
228  Multiply the k-th column of Ta-- / Tb++ with the k-th element of P-/+.
229 *************************************************************************/
230 
231  Maux_a = matcop(Maux_a, Tmm_a);
232  Maux_b = matcop(Maux_b, Tpp_b);
233 
234  for(k = 1; k <= n_beams; k ++)
235  {
236  faux_r = *(Pm->rel+k);
237  faux_i = *(Pm->iel+k);
238 
239  ptr_end=Maux_a->rel+nn_beams;
240  for (ptr_r = Maux_a->rel+k, ptr_i = Maux_a->iel+k;
241  ptr_r <= ptr_end; ptr_r += n_beams, ptr_i += n_beams)
242  cri_mul(ptr_r, ptr_i, *ptr_r, *ptr_i, faux_r, faux_i);
243 
244  faux_r = *(Pp->rel+k);
245  faux_i = *(Pp->iel+k);
246 
247  ptr_end=Maux_b->rel+nn_beams;
248  for (ptr_r = Maux_b->rel+k, ptr_i = Maux_b->iel+k;
249  ptr_r <= ptr_end; ptr_r += n_beams, ptr_i += n_beams)
250  cri_mul(ptr_r, ptr_i, *ptr_r, *ptr_i, faux_r, faux_i);
251  }
252 
253 /*************************************************************************
254  (i) Finish the computation of Tab++ and Tab--:
255  Tpp_ab = Maux_b * Tpp_ab,
256  Tmm_ab = Maux_a * Tmm_ab.
257 
258  (ii) Complete the computation of the matrix product in Rab+- and Rab-+:
259  Rpm_ab = Maux_b * Rpm_ab,
260  Rmp_ab = Maux_a * Rmp_ab.
261 
262 (iii) Finally add the reflection matrix of a single layer to the matrix
263  product in Rab+- and Rab-+:
264 
265  (The minus sign in Rab-+ is due to the negative sign of the matrix
266  product).
267 
268 *************************************************************************/
269 
270 /* (i) */
271  Tpp_ab = matmul(Tpp_ab, Maux_b, Tpp_ab);
272  Tmm_ab = matmul(Tmm_ab, Maux_a, Tmm_ab);
273 
274 /* (ii) */
275  Rpm_ab = matmul(Rpm_ab, Maux_b, Rpm_ab);
276  Rmp_ab = matmul(Rmp_ab, Maux_a, Rmp_ab);
277 
278 /* (iii) */
279 
280 /* Rpm_ab: */
281  ptr_end = Rpm_ab->rel + nn_beams;
282  for(ptr_r = Rpm_ab->rel+1, ptr_i = Rpm_b->rel+1;
283  ptr_r <= ptr_end; ptr_r ++, ptr_i ++)
284  *ptr_r += *ptr_i;
285 
286  ptr_end = Rpm_ab->iel + nn_beams;
287  for(ptr_r = Rpm_ab->iel+1, ptr_i = Rpm_b->iel+1;
288  ptr_r <= ptr_end; ptr_r ++, ptr_i ++)
289  *ptr_r += *ptr_i;
290 
291 /* Rmp_ab: */
292  ptr_end = Rmp_ab->rel + nn_beams;
293  for(ptr_r = Rmp_ab->rel+1, ptr_i = Rmp_a->rel+1;
294  ptr_r <= ptr_end; ptr_r ++, ptr_i ++)
295  *ptr_r = *ptr_i - *ptr_r;
296 
297  ptr_end = Rmp_ab->iel + nn_beams;
298  for(ptr_r = Rmp_ab->iel+1, ptr_i = Rmp_a->iel+1;
299  ptr_r <= ptr_end; ptr_r ++, ptr_i ++)
300  *ptr_r = *ptr_i - *ptr_r;
301 
302 /*************************************************************************
303  - Free temporary storage space,
304  - Write results to output pointers
305  - Return.
306 *************************************************************************/
307

316  *p_Tmm_ab = matcop(*p_Tmm_ab, Tmm_ab);
317  *p_Rpm_ab = matcop(*p_Rpm_ab, Rpm_ab);

325 */
