21 void ftsmooth_usage(FILE *output)
22 {
23  fprintf(output, "\nusage:\n\n%s ", FTSMOOTH_PROG);
24  fprintf(output, "-i <input> -o <output> "
25  "(-m <mode> -c <cutoff> -t <tailoff> ...)\n\n");
26  fprintf(output, "mandatory arguments:\n");
27  fprintf(output, "\n\t-i <input>\tspecify input xy data file to process.\n");
28  fprintf(output, "\t--input <input>\n");
29  fprintf(output, "\n\t-o <output>\tspecify output xy data file.\n");
30  fprintf(output, "\t--output <output>\n");
31  fprintf(output, "\n");
32  fprintf(output, "optional arguments:\n");
33  fprintf(output, "\n\t-c <cut>\tenter a cut off value for fourier transform\n");
34  fprintf(output, "\t--cutoff <cut>\tthis should be between 0 and 1 "
35  "(default=0.5).\n");
36  fprintf(output, "\n\t-d\t\tdelete entries with negative y-values.\n"
37  "\t--delete\n");
38  fprintf(output, "\n\t-h\t\tdisplay syntax help.\n\t--help\n");
39  fprintf(output, "\n\t-m <mode>\tspecify data smoothing operation mode "
40  "(default='n').\n");
41  fprintf(output, "\t--mode <mode>\toptions are: "
42  "'s' smoothing, 'n' no-smoothing.\n");
43  fprintf(output, "\n\t--range <bounds>\n");
44  fprintf(output, "\t-r <bounds>\tthe accepted x-range for the data, "
45  "specified by\n");
46  fprintf(output, "\t\t\t <bounds>. Upper & lower x-axis bounds must be decimals\n");
47  fprintf(output, "\t\t\tand where <bounds> may take the following format:\n");
48  fprintf(output, "\t\t\t\"lb(1):ub(1),lb(2):ub(2),...,lb(N):ub(N)\"\n");
49  fprintf(output, "\t\t\t':' acts as a seperator between lb & ub\n");
50  fprintf(output, "\t\t\t',' acts as a seperator between indicies\n");
51  fprintf(output, "\t\t\t'i' or '*' will ignore bounds, i.e.\n");
52  fprintf(output, "\t\t\t'0.2:10,0.5:*' will output data between 0.2 & max\n");
53  fprintf(output, "\t\t\te.g. \"-r i 300\" will only set the upper bound to 300.\n");
54  fprintf(output, "\n\t-t <tail>\tenter a tail-off value for fourier transform\n");
55  fprintf(output, "\t\t\t(default=10).\n");
56  fprintf(output, "\t--tailoff <tail>\n");
57  fprintf(output, "\n\t--version\tprint version info.\n");
58  fprintf(output, "\n\t-y (<offset>)\toffset the y-axis data by the specified value\n");
59  fprintf(output, "\t--yoffset\tnote: if no value is specified, the offset will\n");
60  fprintf(output, "\t\t\tbe minima = 0.\n");
61  fprintf(output, "\n");
62 }
